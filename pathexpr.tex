\begin{definition}
  The set of \emph{regular expressions} over $\Sigma$, written
  $RegExp(\Sigma)$, is the smallest set such that
  $0, 1 \in RexExp(\Sigma)$, $\Sigma \subseteq RegExp(\Sigma)$,
  and $(R_1 + R_2), (R_1 \cdot R_2), R^* \in RegExp(\Sigma)$ for
  all $R, R_1, R_2 \in RegExp(\Sigma)$.

  $SubExp(R)$ for $R \in RegExp(\Sigma)$ is the set of
  subexpressions of $R$, including $R$ itself.
\end{definition}

\begin{definition}
  % taken more or less verbatim from Algebraic Program Analysis +
  % Fast Algorithms for Solving Path Problems
  The \emph{standard interpretation} is the function
  $\si{\cdot}: RegExp(\Sigma) \to \textbf{L}$, where $\textbf{L}$
  is the set of regular languages, such that
  $\si{0} = \emptyset$, $\si{1} = \{\epsilon\}$ where $\epsilon$
  is the empty string, $\si{a} = \{a\}$ for all $a \in \Sigma$,
  and
  \begin{align*}
  \si{R_1 + R_2} &= \si{R_1} \cup \si{R_2} \\
  \si{R_1 \cdot R_2} &= \{x_1x_2| x_1 \in \si{R_1}, x_2 \in
                       \si{R_2}\} \\
  \si{R^*} &= \{x_1x_2\dots x_n| x_1,x_2,\dots,x_n \in \si{R}\}
  \end{align*}
  for all $R, R_1, R_2 \in RegExp(\Sigma)$.
\end{definition}

\begin{definition}
  For any function $wt: \Sigma \to \mathbb{Z}$, the \emph{length}
  of a regular expression is given by
  $L: RegExp(\Sigma) \to \mathbb{Z} \cup \{-\infty\}$ where
  $L(0) = -\infty$, $L(1) = L(R^*) = 1$, $L(a) = wt(a)$ for all
  $a \in \Sigma$, $L(R_1 + R_2) = \max\{L(R_1), L(R_2)\}$, and
  $L(R_1 \cdot R_2) = L(R_1) + L(R_2)$. 
\end{definition}

Let $X$ and $X'$ be sets of variables such that for each
$x\in X$, there is a corresponding $x' \in X'$. As an abuse of
notation, we use $X = X' \equiv \bigwedge_{x \in X} x =
x'$. $Vars(F)$ for a formula $F$ is the set of free variables in
$F$. $Formula(X, X')$ denotes the set of formulas such that
$Vars(F) \subseteq X \cup X'$. 

\begin{definition}
  A \emph{control flow graph} $G = \tuple{V, E, r, F}$ is a graph
  with vertices $V$, edges $E$, a root vertex $r$ of indegree 0
  from which every vertex $v \in V$ is reachable, and a function
  $F: E \to Formula(X, X')$ for some $X, X'$. We use
  $F_{\tuple{u,v}}(Y, Y')$ to denote
  $F(\tuple{u,v})[X \mapsto Y, X' \mapsto Y']$. We let
  $Paths_G(u,v)$ be the set of all (possibly non-simple) paths
  from $u$ to $v$ in $G$.
\end{definition}

\begin{definition}
  A \emph{path expression} $P$ of type $\tuple{u, v}$ on
  $G = \tuple{V, E, r, F}$ is a regular expression over
  $E$ such that $\si{P} \subseteq Paths_G(u,v)$ for some
  $u,v \in V$. We denote the set of path expressions on $G$ by
  $PathExp(G)$.
\end{definition}

Suppose $star$ is a function that takes a formula $F$ and the
pair of variable vocabulairies $X$ and $X'$ that $F$ relates and
returns some overapproximation of the reflexive transitive
closure of $F$ over $X$ and $X'$ containing no other free
variables.

\begin{definition}
  The \emph{transition formula interpretation}
  is the function $\tf{\cdot}: PathExp(G) \to Formula(X, X')$
  where $\tf{0} \equiv false$, $\tf{1} \equiv X = X'$,
  $\tf{e} \equiv F_e(X, X')$ for all $e \in E$, and
  \begin{align*}
    \tf{P_a + P_b} &\equiv \tf{P_a} \lor \tf{P_b} \\
    \tf{P_a \cdot P_b} &\equiv \exists X''. \tf{P_a}(X, X'')
                         \land \tf{P_b}(X'', X') \\
    \tf{P^*} &\equiv star(\tf{P}, \tuple{X, X'}) 
  \end{align*}
  for all $P, P_a, P_b \in PathExp(G)$. 

  We use $\tf{P}(Y, Y')$ to denote
  $\tf{P}[X \mapsto Y, X' \mapsto Y']$.
\end{definition}

\begin{definition}
  Suppose there are an infinite set of variable vocabularies
  $\{X_{-\infty}, X_0, X_1, X_2, \dots\}$, and suppose that $L$
  is taken with respect to the function $wt: E \to \mathbb{Z}$
  that maps every edge to 1. Then, define $I$ mapping path
  expressions and integer lower and upper bounds to formulas
  where $I(0, l, u) \equiv false$, $I(1, l, u) \equiv X_l = X_u$,
  $I(e, l, u) \equiv F_e(X_l, X_u)$ for all $e \in E$, and
  \begin{align*}
    I(P_a + P_b, l, u)
    &\equiv I(P_a, l, u) \lor I(P_b, l, u) \\
    I(P_a \cdot P_b, l, u)
    &\equiv I(P_a, l, l + L(P_a)) \land I(P_b, l + L(P_a), u) \\
    I(P^*, l, u)
    &\equiv star(I(P, l+1, l+1+L(P)), \tuple{X_{l+1},
      X_{u+1}})[X_{l+1} \mapsto X_l, X_{l+1+L(P)} \mapsto X_r]
  \end{align*}
  for all $P, P_a, P_b \in PathExp(G)$. 
\end{definition}

\begin{lemma}
  \label{pe:freevars}
  For all $P \in PathExp(G)$, for all $0 \le l < l + L(P) \le u$,
  $Vars(I(P, l, u)) \subseteq \bigcup_{i = l}^u X_i$.
\end{lemma}

\begin{proof}
  The claim holds trivially for $P = 0$, $P = 1$, $P = P'^*$, or
  $P = \tuple{v, w}$. Suppose the claim holds for $P_a$ and
  $P_b$. Clearly, for all $0 \le l < l + L(P_a + P_b) \le u$,
  $Vars(I(P_a + P_b, l, u)) \subseteq Vars(I(P_a, l, u)) \cup
  Vars(I(P_b, l, u)) = \bigcup_{i = l}^u X_i$. And, since
  $L(P_a \cdot P_b) = L(P_a) + L(P_b)$, for all
  $0 \le l < l + L(P_a \cdot P_b) \le u$
  $Vars(I(P_a \cdot P_b, l, u)) \subseteq Vars(I(P_a, l, u +
  L(P_a))) \cup Vars(I(P_b, l + L(P_a), u)) = (\bigcup_{i = l}^{l
    + L(P_a)} X_i) \cup (\bigcup_{i = l + L(P_a)}^u X_i) =
  \bigcup_{i = l}^u X_i$. By induction, the claim must hold for
  all $P$.
\end{proof}

\begin{lemma}
  \label{pe:zero}
  If $L(P) = -\infty$, then for all $l, u$,
  $\tf{P} \equiv I(P, l, u) \equiv false$.
\end{lemma}

\begin{proof}
  We proceed by structural induction on $P$. If $P = 0$, then the
  claim holds trivially, and if $P = 1$ or $P = P'^*$, then the
  claim holds vacuously. Suppose the claim holds for $P_a$ and
  $P_b$ and suppose that $L(P_a + P_b) = -\infty$. Then,
  $L(P_a) = L(P_b) = -\infty$. So,
  $\tf{P_a + P_b} \equiv \tf{P_a} \lor \tf{P_b} \equiv false
  \equiv I(P_a, l, u) \lor I(P_b, l, u) \equiv I(P_a + P_b, l,
  u)$. Similarly, suppose $L(P_a \cdot P_b) = -\infty$. Then,
  either $L(P_a) = -\infty$ or $L(P_b) = -\infty$. Either way,
  $\tf{P_a \cdot P_b} \equiv \exists X''. \tf{P_a}(X, X'') \land
  \tf{P_b}(X'', X') \equiv false \equiv I(P_a, l, l + L(P_a))
  \land I(P_b, l + L(P_a), u) \equiv I(P_a \cdot P_b, l, u)$. 
\end{proof}

\begin{theorem}
  For all $P \in PathExp(G)$,
  $Sat(\tf{P}) \iff Sat(I(P, 0, L(P)))$.
\end{theorem}

\begin{proof}
  As an abuse of notation, let $\exists S$ for a set
  $S = \{x_1, \dots, x_m\}$ denote $\exists x_1, \dots,
  x_m$. Then, we proceed by structural induction on $P$ with the
  hypothesis that for all $0 \le l < l + L(P) \le u$,
  $\tf{P} \equiv \exists Vars(I(P, l, u)). I(P, l, u) \land X =
  X_l \land X' = X_r$. Note that this is sufficient by Lemma
  \ref{pe:zero}. If $P = 0$, $P = 1$, or $P = \tuple{v, w}$, then
  the claim holds trivially.

  Suppose the claim holds for $P_a$ and $P_b$ and take arbitrary
  $0 \le l < l + L(P_a + P_b) \le u$. Let
  $V_a = Vars(I(P_a, l, u))$, $V_b = Vars(I(P_b, l, u))$, and
  $V = Vars(I(P_a + P_b, l, u))$. We know by Lemma
  \ref{pe:freevars} that $ V_a \cup V_b \subseteq V$. Then,
  \begin{align*}
    \tf{P_a + P_b}
    &\equiv \tf{P_a} \lor \tf{P_b} \\
    &\equiv
      (\exists V_a. I(P_a, l, u) \land X = X_l \land X' = X_r)
      \lor (\exists V_b. I(P_b, l, u) \land X = X_l \land X' =
      X_r) \\
    &\equiv
      \exists V. I(P_a + P_b, l, u) \land X = X_l \land X' = X_r
  \end{align*}
  by the induction hypothesis and (if applicable) Lemma
  \ref{pe:zero}. % todo: deal with 0 case more effectively

  Now, take arbitrary take arbitrary
  $0 \le l < l + L(P_a \cdot P_b) \le u$. Let
  $W_a = Vars(I(P_a, l, l + L(P_a)))$,
  $W_b = Vars(I(P_b, l + L(P_a), u))$, and
  $W = Vars(I(P_a \cdot P_b, l, u))$. By Lemma \ref{pe:freevars},
  we know that $W_a \cup W_b \subseteq W$, but also that
  $W_a \cap W_b = X_{l + L(P_a)}$. Furthermore,
  $L(P_a), L(P_b) \ge 0$. So,
  \begin{align*}
    \tf{P_a \cdot P_b}
    &\equiv \exists X''.
      \tf{P_a}(X, X'') \land \tf{P_b}(X'', X') \\
    &\equiv \exists X''.
      (\exists W_a. I(P_a, l, l + L(P_a)) \land
      X = l \land X'' = X_{l + L(P_a)}) \\
    & \qquad \land (\exists W_b. I(P_a, l + L(P_a), u) \land
      X'' = X_{l + L(P_a)} \land X' = u) \\
    &\equiv \exists X''.
      \exists W.
      I(P_a, l, l + L(P_a)) \land I(P_a, l + L(P_a), u) \land
      X = l \land X'' = X_{l + L(P_a)} \land X' = u \\
    &\equiv \exists W.
      I(P_a \cdot P_b, l, u) \land X = l \land X' = u
  \end{align*}

  Finally, suppose the claim holds for $P$ and take arbitrary
  $0 \le l < l + 1 \le u$. Obviously,
  $0 \le l + 1 < l + 1 + L(P)$, so by the induction hypothesis,
  $\tf{P} \equiv \exists Vars(I(P, l+1, l+1+L(P))). I(P, l+1,
  l+1+L(P)) \land X = X_{l+1} \land X' = X_{l+1+L(P)}$. So,
  $\tf{P}(X_{l+1}, X_{l+1+L(P)}) \equiv star(I(P, l+1, l+1+L(P)),
  \tuple{X_{l+1}, X_{l+1+L(P)}})$. Therefore,
  $\tf{P^*}(X_l, X_u) \equiv I(P^*, l, u)$, and the claim holds.
  
\end{proof}

\begin{definition}
  In a control flow graph $G = \tuple{V, E, r, F}$, $v\in V$ is
  said to \emph{dominate} $u \in V$ if every path from $r$ to $u$
  in $G$ goes through $v$. If $u \ne v$, then $v$ is said to
  \emph{strictly dominate} $u$. If $v$ strictly dominates $u$ and
  is dominated by every vertex that dominates $u$, then $v$ is
  the \emph{immediate dominator} of $u$, denoted $v =
  idom(u)$. Note that there is a rooted tree whose vertices are
  $V$, whose root is $r$, and whose edges $\tuple{v, u}$
  correspond to $v = idom(u)$. We let $children(v)$ be the set of
  children of $v$ in this tree, i.e. $\{u|v=idom(u)\}$. $E|_v$
  denotes the set of edges $\tuple{w_1, w_2}$ where $w_2$ is
  strictly dominated by $v$. 
\end{definition}

\begin{definition}
  A control flow graph $G = \tuple{V, E, r, F}$ is reducible if
  every cycle in $G$ contains an edge $\tuple{u, v}$ where $v$
  dominates $u$. Intuitively, every cycle has a natural
  entrypoint. 
\end{definition}

\begin{algorithm}[H]
  \caption{Tarjan's Algorithm (on Reducible Flow Graphs)}
  \label{alg:tarjan}

  \SetAlgoLined
  % \SetAlgoNoEnd
  
  \KwIn{A reducible control flow graph $G=\tuple{V, E, r, F}$}
  
  \KwOut{A function $pe: V \to PathExp(G)$ such that for all
    $v \in V$, $pe(v)$ is a well-formed path expression
    recognizing exactly $Paths_G(r, v)$}

  Initialize compressed weighted forest on $V$ \;

  $children \gets$ the dominator tree of $G$ \;

  $SolveReducible(r)$ \;

  \Return $\lambda v. eval(v)$ \;

  \BlankLine

  \SetKwProg{Subroutine}{Subroutine}{ is}{end}

  \Subroutine{SolveReducible(v)}{

    \KwIn{A vertex $v$ such that $eval(v) = 1$}

    \KwResult{For all $u$ dominated by $v$,
      $\si{eval(u)} = Paths_G(r, u) \cap E|_v^*$}

    \ForEach{$u \in children(v)$ in topological order of the
      sibling graph}{
      
      $SolveReducible(u)$ \;

      $cycle, noncycle \gets 0$ \;

      \ForEach{$w \in pred(u)$}{
        \lIf{$u$ dominates $w$}{
          $cycle \gets cycle + eval(w) \cdot \tuple{w, u}$
        }
        \lElse{
          $noncycle \gets noncycle + eval(w) \cdot \tuple{w, u}$
        }
      }

      $link(u, noncycle \cdot cycle^*, v)$ \;
      
    }

  }

\end{algorithm}

\begin{algorithm}[H]
  \caption{Gaussian Elimination}
  \label{alg:gauss}

  \SetAlgoLined
  % \SetAlgoNoEnd
  
  \KwIn{A control flow graph $G=\tuple{V, E, r, F}$ with vertices
    labeled $1, \dots, n-1, n = r$}
  
  \KwOut{A function $pe: V \to PathExp(G)$ such that for all
    $v \in V$, $pe(v)$ is a well-formed path expression
    recognizing exactly $Paths_G(r, v)$}

  \tcc{$\si{P(u, v)} \subseteq Paths_G(u, v)$ for all
    $u, v \in V$}

  $P \gets \lambda \tuple{u, v}. 0$ \;

  \lForEach{$\tuple{u, v} \in E$}{
    $P(u, v) \gets P(u, v) + \tuple{u, v}$
  }

  \For{$v = 1$ \KwTo $n-1$}{
    \For{$u = v + 1$ \KwTo $n$}{
      $P(u, v) \gets P(u, v) \cdot P(v, v)^*$ \;
      \For{$w = 1$ \KwTo $n$, $w \ne v$}{
        $P(u, w) \gets P(u, w) + P(u, v) \cdot P(v, w)$ \;
      }
    }
  }

  $pe \gets \lambda v. P(r, v)$ \;

  \Return $pe$ \;

\end{algorithm}

\begin{lemma}
  After the $v$th iteration of the outer loop, for all
  $u, w \in V$, $P(u, w)$ is a well-formed path expression
  recognizing exactly the paths from $u$ to $w$ in $G$ whose
  intermediate vertices are smaller than $\min\{u, v+1\}$. 

\end{lemma}

  We proceed by induction on $v$, and assume that the claim holds
  before the $v$th iteration. This is clearly true before the
  first iteration since all path expressions are either empty or
  edges. Consider any $u < v$; by the structure of the for loops,
  $P(u, w)$ for all $w$ is unchanged during this iteration, and
  $\min\{u, v+1\} = \min\{u, v\} = u$, so the claim holds at the
  end of this iteration. Similarly, for all $w$, $P(v, w)$ is
  left unchanged, and we know by the induction hypothesis the
  intermediate vertices in the paths it recognizes are smaller
  than $\min \{v, v\} = v$; ergo, the claim holds for such $w$ at
  the end of this iteration.

  Therefore, it only remains to prove the claim for any $u >
  v$. Note that $\min\{u, v\} = v$ and $\min\{u, v+1\} =
  v+1$. First, the only assignment to $P(u, v)$ this iteration is
  to concatenate $P(v, v)^*$ to it. Consider any path from $u$ to
  $v$ whose intermediate vertices are smaller than $v+1$. It must
  be the concatenation of a non-empty path $p_0$ from $u$ to $v$
  whose intermediate vertices are smaller than $v$ and $k$ paths
  $p_1, \dots, p_k$ from $v$ to $v$ whose intermediate vertices
  are smaller than $v$. By the induction hypothesis, $p_0$ is
  recognized by the previous $P(u, v)$, and each $p_i$ for
  $1 \le i \le k$ is recognized by $P(v, v)$. Thus, after this
  iteration $P(u, v)$ recognizes exactly the paths from $u$ to
  $v$ in $G$ whose intermediate vertices are smaller than $v+1$.

  Consider any $w \ne v$ and any path from $u$ to $w$ whose
  intermediate vertices are smaller than $v+1$. Either it is a
  path whose intermediate vertices are smaller than $v$, or it is
  the concatenation of a non-empty path $p_1$ from $u$ to $v$
  whose intermediate vertices are smaller than $v+1$ and a
  non-empty path $p_2$ from $v$ to $w$ whose intermediate
  vertices are smaller than $v$. Thus, $P(u, w) + P(u, v) \cdot
  P(v, w)$ recognizes exactly these paths by the induction
  hypothesis and our previous analysis of $P(u, v)$.
\end{proof}
