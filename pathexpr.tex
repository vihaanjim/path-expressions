\begin{definition}
  The set of \emph{regular expressions} over $\Sigma$, written
  $RegExp(\Sigma)$, is the smallest set such that
  $0, 1 \in RexExp(\Sigma)$, $\Sigma \subseteq RegExp(\Sigma)$,
  and $(R_1 + R_2), (R_1 \cdot R_2), R^* \in RegExp(\Sigma)$ for
  all $R, R_1, R_2 \in RegExp(\Sigma)$.

  $SubExp(R)$ for $R \in RegExp(\Sigma)$ is the set of
  subexpressions of $R$, including $R$ itself.
\end{definition}

\begin{definition}
  % taken more or less verbatim from Algebraic Program Analysis +
  % Fast Algorithms for Solving Path Problems
  The \emph{standard interpretation} is the function
  $\si{\cdot}: RegExp(\Sigma) \to \textbf{L}$, where $\textbf{L}$
  is the set of regular languages, such that
  $\si{0} = \emptyset$, $\si{1} = \{\epsilon\}$ where $\epsilon$
  is the empty string, $\si{a} = \{a\}$ for all $a \in \Sigma$,
  and
  \begin{align*}
  \si{R_1 + R_2} &= \si{R_1} \cup \si{R_2} \\
  \si{R_1 \cdot R_2} &= \{x_1x_2| x_1 \in \si{R_1}, x_2 \in
                       \si{R_2}\} \\
  \si{R^*} &= \{x_1x_2\dots x_n| x_1,x_2,\dots,x_n \in \si{R}\}
  \end{align*}
  for all $R, R_1, R_2 \in RegExp(\Sigma)$.
\end{definition}

Let $X$ and $X'$ be sets of variables such that for each
$x\in X$, there is a corresponding $x' \in X'$. As an abuse of
notation, we use $X = X' \equiv \bigwedge_{x \in X} x =
x'$. $Formula(X, X')$ denotes the set of formulas with all free
variables in either $X$ or $X'$. 

\begin{definition}
  A \emph{control flow graph} $G = \tuple{V, E, r, F}$ is a graph
  with vertices $V$, edges $E$, a root vertex $r$ of indegree 0
  from which every vertex $v \in V$ is reachable, and a function
  $F: E \to Formula(X, X')$ for some $X, X'$. We use
  $F_{\tuple{u,v}}(Y, Y')$ to denote
  $F(\tuple{u,v})[X \mapsto Y, X' \mapsto Y']$. We let
  $Paths_G(u,v)$ be the set of all (possibly non-simple) paths
  from $u$ to $v$ in $G$.

\end{definition}

\begin{definition}
  A \emph{path expression} $P$ of type $\tuple{u, v}$ on
  $G = \tuple{V, E, r, F}$ is a regular expression over
  $E$ such that $\si{P} \subseteq Paths_G(u,v)$ for some
  $u,v \in V$. We denote the set of path expressions on $G$ by
  $PathExp(G)$.

\end{definition}

Suppose $star$ is a function that takes a formula and the pair of
variable vocabulairies that it relates and returns some
overapproximation of the reflexive transitive closure of the
formula over those vocabularies, containing no other free
variables.

\begin{definition}
  The \emph{transition formula interpretation}
  is the function $\tf{\cdot}: PathExp(G) \to Formula(X, X')$
  where $\tf{0} \equiv false$, $\tf{1} \equiv X = X'$,
  $\tf{e} \equiv F_e(X, X')$ for all $e \in E$, and
  \begin{align*}
    \tf{P_a + P_b} &\equiv \tf{P_a} \lor \tf{P_b} \\
    \tf{P_a \cdot P_b} &\equiv \exists X''. \tf{P_a}(X, X'')
                         \land \tf{P_b}(X'', X') \\
    \tf{P^*} &\equiv star(\tf{P}, \tuple{X, X'}) 
  \end{align*}
  for all $P, P_a, P_b \in PathExp(G)$. 

  We use $\tf{P}(Y, Y')$ to denote
  $\tf{P}[X \mapsto Y, X' \mapsto Y']$.
\end{definition}

\begin{definition}
  % todo: better name
  The \emph{loop-free subexpressions} of $P$ are given by $LF(P)$
  where
  \begin{align*}
    LF(1) &= \{1\} \\
    LF(\tuple{u,v}) &= \{\tuple{u,v}\} \\
    LF(P_a + P_b) &= LF(P_a) \cup LF(P_b) \cup \{P_a + P_b\} \\
    LF(P_a \cdot P_b) &= LF(P_a) \cup LF(P_b) \cup \{P_a \cdot
                        P_b\} \\
    LF(P^*) &= \{P^*\} 
  \end{align*}

  Note that $P \in LF(P)$ for all $P$.
\end{definition}

\begin{definition}
  Suppose every vertex $v \in V$ is associated with variable
  vocabularies $X_v$ and $X'_v$. Then, a path expression $P$ is
  \emph{well-formed} if the following hold:
  \begin{itemize}
  \item $0, 1 \not \in SubExp(P)$
  \item For all $P_a \cdot P_b \in P$,
    $LF(P_a) \cap LF(P_b) = \emptyset$.    
  \item There exist functions $L$ and $R$ mapping $SubExp(P)$ to
    variable vocabularies such that
    $L(\tuple{u,v}) \in \{X_u, X'_u\}$ and
    $R(\tuple{u,v}) \in \{X_v, X'_v\}$;
    $L(P_a + P_b) = L(P_a) = L(P_b)$ and
    $R(P_a + P_b) = R(P_a) = R(P_b)$;
    $L(P_a \cdot P_b) = L(P_a)$, $R(P_a \cdot P_b) = R(P_b)$, and
    $R(P_a) = L(P_b)$ for $P_a \cdot P_b \in SubExp(P)$; and
    $L({P_a}^*) = R(P_a)$ and $R({P_a}^*) = L(P_a)$.
  \end{itemize}
\end{definition}

\begin{algorithm}[H]
  \caption{Path Expression DAG to Transition Formula}
  \label{alg:path-exp-to-tf}

  \SetAlgoLined
  % \SetAlgoNoEnd
  
  \KwIn{A control flow graph $G=\tuple{V, E, r, F}$ and a
    well-formed path expression $P$ whose distinct subexpressions
    $\{P_1,\dots,P_n\}$ are labeled so that
    $P_i \in SubExp(P_j) \implies i < j$.}
  
  \KwOut{A formula $F$ such that
    $Sat(\tf{P}) \iff Sat(F)$}

  \ForEach{$i = 1$ \KwTo $n$}{
    \Switch{$P_i$}{
      \lCase{$\tuple{u, v} \in E$}{
        $F_i \gets F_e(L(P_i), R(P_i))$
      }

      \lCase{$P_j + P_k$}{
        $F_i \gets F_j \lor F_k$
      }

      \lCase{$P_j \cdot P_k$}{
        $F_i \gets F_j \land F_k$
      }

      \Case{${P_j}^*$}{

        $F^* \gets star(F_j, \tuple{L(P_j), R(P_j)})$ \;

        $F_i \gets F^*[L(P_j) \mapsto L(P_i), R(P_j) \mapsto
        R(P_i)]$ \;
        
      }
    }
  }

  $F \gets F_n$ \;

  \Return $F$ \;

\end{algorithm}

\begin{lemma}
  \label{pe:freevars}
  For all $1 \le i \le n$, all free variables in $F_i$ are
  members of $\bigcup_{P_a \in LF(P_i)} (L(P_a) \cup R(P_a))$.
\end{lemma}

\begin{proof}
  We proceed by induction on $i$ and assume the claim holds for
  each $l < i$. This is trivially the case for $i = 1$. For
  $l = i$, if $P_i = e \in E$, or $P_i = {P_k}^*$ for some $k$,
  then the only free variables in $F_i$ are in $L(P_i)$ and
  $R(P_i)$, so we are done. Otherwise, $P_i = P_j + P_k$ or
  $P_i = P_j \cdot P_k$ for some $j$ and $k$, and the free
  variables in $F_i$ are exactly those in $F_j$ or $F_k$. By the
  induction hypothesis and the definition of $LF$, the claim
  holds for $i$.
\end{proof}

Let $V_j = \bigcup_{P_a \in LF(P_j)} (L(P_a) \cup R(P_a))$. By
definition, if $P_j \in LF(P_i)$, then $V_j \subseteq V_i$. As an
abuse of notation, we write $\exists V_j$ to denote
$\exists x_1, \dots, x_m$, where $V_j = \{x_1, \dots, x_m\}$.

\begin{theorem}
  $Sat(\tf{P}) \iff Sat(F)$
\end{theorem}

\begin{proof}
  We proceed by induction on the iteration $i$ of the for loop,
  with the hypothesis that before the $i$th iteration, for all
  $j < i$,
  $\tf{P_j} \equiv \exists V_j. F_j \land X = L(P_j) \land X' =
  R(P_j)$. Since $F \equiv F_n$, and since $X$ and $X'$ appear
  nowhere else in $F$ by Lemma \ref{pe:freevars}, this is
  sufficient to prove the theorem.

  Certainly this is true before the first iteration. Suppose this
  is true before the $i$th iteration of the loop. Clearly, the
  claim still holds for $j < i$ after this iteration since $F_j$
  is not modified. For $j = i$, let us consider each possible
  form of $P_i$:

  \begin{itemize}
  \item $P_i = \tuple{u, v}$: We have that
    $\tf{\tuple{u, v}} = F_{\tuple{u, v}}(X, X')$ and
    $F_i \equiv F_{\tuple{u, v}}(L(P_i), R(P_i))$.  So,
    $\exists V_i. F_i \land X = L(P_i) \land X' = R(P_i) \equiv
    F_{\tuple{u, v}}(X, X') \equiv \tf{P_i}$.
    
  \item $P_i = P_j + P_k$:
    $\tf{P_j+P_k} \equiv \tf{P_j} \lor \tf{P_k}$. By the
    induction hypothesis,
    $\tf{P_i} \equiv (\exists V_j. F_j \land X = L(P_j) \land X'
    = R(P_j)) \lor (\exists V_k. F_k \land X = L(P_k) \land X' =
    R(P_k))$.

    Since $V_i = V_j \cup V_k$ by definition, and since
    $L(P_i) = L(P_j) = L(P_k)$ and $R(P_i) = R(P_j) = R(P_k)$ we
    have that
    $\tf{P_i} \equiv \exists V_i. (F_j \lor F_k) \land X = L(P_i)
    \land X' = R(P_i)$, as desired.
    
  \item $P_i = P_j \cdot P_k$:
    $\tf{P_j \cdot P_k} \equiv \exists X''_1, X''_2. \tf{P_j}(X,
    X''_1) \land \tf{P_k}(X''_2, X') \land X''_1 = X''_2$. So, by
    the induction hypothesis,
    $\tf{P_i} \equiv \exists X''_1, X''_2. (\exists V_j. F_j
    \land X = L(P_j) \land X''_1 = R(P_j)) \land (\exists
    V_k. F_k \land X''_2 = L(P_k) \land X' = R(P_k)) \land X''_1
    = X''_2$.

    By well-formedness, $LF(P_j) \cap LF(P_k) = \emptyset$. So,
    by Lemma \ref{pe:freevars}, there are no free variables
    present in both $F_j$ and $F_k$. Therefore, since
    $L(P_i) = L(P_j)$, $R(P_i) = R(P_k)$, and $R(P_j) = L(P_k)$,
    \begin{align*}
      \tf{P_i}
      &\equiv \exists X''_1, X''_2. \exists V_i. F_j \land
        X = L(P_j) \land X''_1 = R(P_j) \land F_k \land
        X''_2 = L(P_k) \land X' = R(P_k) \land X''_1 = X''_2 \\
      &\equiv \exists V_i. F_j \land X = L(P_j) \land
        R(P_j) = L(P_k) \land F_k \land X' = R(P_k) \\
      & \equiv \exists V_i. F_i \land X = L(P_i) \land X' = R(P_i)
    \end{align*}

  \item $P_i = {P_j}^*$: We know by the induction hypothesis that
    $\tf{P_j} \equiv \exists V_j. F_j \land X = L(P_j) \land X' =
    R(P_j)$.  So,
    $F^* = star(F_j, \tuple{L(P_j), R(P_j)}) \equiv
    \tf{P_i}(L(P_j), R(P_j))$. Therefore, for
    $F_i = F^*[L(P_j) \mapsto L(P_i), R(P_j) \mapsto R(P_i)]$,
    $\exists V_i. F_i \land X = L(P_i) \land X' = R(P_i) \equiv
    \tf{P_i}$.
  \end{itemize}
\end{proof}

\begin{algorithm}[H]
  \caption{Path Expression Finding}
  \label{alg:gauss}

  \SetAlgoLined
  % \SetAlgoNoEnd

  \SetKw{KwDownTo}{downto}
  
  \KwIn{A control flow graph $G=\tuple{V, E, r, F}$ with vertices
    labeled $1 = r, 2, \dots, n$}
  
  \KwOut{A function $pe: V \to PathExp(G)$ such that for all
    $v \in V$, $pe(v)$ is a well-formed path expression
    recognizing exactly $Paths_G(r, v)$}

  \tcc{$\si(P(u, v)) \subseteq Paths_G(u, v)$}
  
  $P \gets \lambda \tuple{u,v}. 0$ \;

  \For{$i = 1$ \KwTo $n-1$}{
    \For{$j = i + 1$ \KwTo $n$}{
      \For{$k = i + 1$ \KwTo $n$}{
        $P(j, k) \gets P(j, k) + P(j, i) \cdot P(i, i)^* \cdot
        P(i, k)$ \;
      }
    }
  }

  % B*(C\Delta*DB*)*

  $P(n, n) \gets P(n, n)^*$ \;

  \For{$i = n - 1$ \KwDownTo $1$}{

    $X \gets 0$ \;

    $C', D' \gets \lambda v. 0$ \;
    
    \For{$j = i + 1$ \KwTo $n$}{
      \For{$k = i + 1$ \KwTo $n$}{
        $C'(k) \gets C'(k) + P(i, j) \cdot P(j, k)$ \;
        $D'(j) \gets D'(j) + P(j, k) \cdot P(k, i)$ \;
        $X \gets X + P(i, j) \cdot P(j, k) \cdot P(k, i)$ \;
      }
    }

    \For{$j = i + 1$ \KwTo $n$}{
      $P(i, j) \gets P(i, i)^* \cdot C'(j)$ \;
      $P(j, i) \gets D'(j) \cdot P(i, i)^*$ \;
    }

    $P(i, i) \gets P(i, i)^* \cdot (X \cdot P(i, i)^*)^*$ \;
  }
  
  $pe \gets \lambda v. P(r, v)$

  \Return $pe$ \;
\end{algorithm}
