\begin{definition}
  A \emph{regular expression} over $\Sigma$ is an element of the
  set $RegExp(\Sigma)$, which is the smallest set such that
  $0, 1 \in RexExp(\Sigma)$, $\Sigma \subseteq RegExp(\Sigma)$,
  and $(R_1 + R_2), (R_1 \cdot R_2), R^* \in RegExp(\Sigma)$ for
  all $R, R_1, R_2 \in RegExp(\Sigma)$.

  $SubExp(R)$ for $R \in RegExp(\Sigma)$ is the set of
  subexpressions of $R$, including $R$ itself. $R$ is
  \emph{simple} if $R = 0$ or $0 \not\in SubExp(R)$. $R$ is
  \emph{non-empty} if $R \ne 0$.
\end{definition}

\begin{definition}
  % taken more or less verbatim from Algebraic Program Analysis +
  % Fast Algorithms for Solving Path Problems
  The \emph{standard interpretation} is the function
  $\si{\cdot}: RegExp(\Sigma) \to \textbf{L}$, where $\textbf{L}$
  is the set of regular languages, such that
  $\si{0} = \emptyset$, $\si{1} = \{\epsilon\}$ where $\epsilon$
  is the empty string, $\si{a} = \{a\}$ for all $a \in \Sigma$,
  and
  \begin{align*}
  \si{R_1 + R_2} &= \si{R_1} \cup \si{R_2} \\
  \si{R_1 \cdot R_2} &= \{x_1x_2| x_1 \in \si{R_1}, x_2 \in
                       \si{R_2}\} \\
  \si{R^*} &= \{x_1x_2\dots x_n| x_1,x_2,\dots,x_n \in \si{R}\}
  \end{align*}
  for all $R, R_1, R_2 \in RegExp(\Sigma)$.
\end{definition}

Let $X$ and $X'$ be sets of variables such that for each
$x\in X$, there is a corresponding $x' \in X'$. As an abuse of
notation, we use $X = X' \equiv \bigwedge_{x \in X} x =
x'$. $Formula(X, X')$ denotes the set of formulas with all free
variables in either $X$ or $X'$. 

\begin{definition}
  A \emph{control flow graph} $G = \tuple{V, E, r, F}$ is a graph
  with vertices $V$, edges $E$, a root vertex $r$ of indegree 0
  from which every vertex $v \in V$ is reachable, and a function
  $F: E \to Formula(X, X')$ for some $X, X'$. We use
  $F_{\tuple{u,v}}(Y, Y')$ to denote
  $F(\tuple{u,v})[X \mapsto Y, X' \mapsto Y']$. We let
  $Paths_G(u,v)$ be the set of all (possibly non-simple) paths
  from $u$ to $v$ in $G$.

\end{definition}

\begin{definition}
  A \emph{path expression} $P$ of type $\tuple{u, v}$ on
  $G = \tuple{V, E, r, F}$ is a simple regular expression over
  $E$ such that $\si{P} \subseteq Paths_G(u,v)$ for some
  $u,v \in V$. We denote the set of path expressions on $G$ by
  $PathExp(G)$.

\end{definition}

Suppose $star$ is a function that takes a formula and the pair of
variable vocabulairies that it relates and returns some
overapproximation of the reflexive transitive closure of the
formula over those vocabularies, containing no other free
variables.

\begin{definition}
  The \emph{transition formula interpretation}
  is the function $\tf{\cdot}: PathExp(G) \to Formula(X, X')$
  where $\tf{0} \equiv false$, $\tf{1} \equiv X = X'$,
  $\tf{e} \equiv F_e(X, X')$ for all $e \in E$, and
  \begin{align*}
    \tf{P_1 + P_2} &\equiv \tf{P_1} \vee \tf{P_2} \\
    \tf{P_1 \cdot P_2} &\equiv \exists X''. \tf{P_1}(X, X'')
                         \wedge \tf{P_2}(X'', X') \\
    \tf{P^*} &\equiv star(\tf{P}, \tuple{X, X'}) 
  \end{align*}
  for all $P, P_1, P_2 \in PathExp(G)$. 

  We use $\tf{P}(Y, Y')$ to denote
  $\tf{P}[X \mapsto Y, X' \mapsto Y']$.
\end{definition}

\begin{definition}
  The \emph{loop-free subexpressions} of $P$ are given by $LF(P)$
  where
  \begin{align*}
    LF(1) &= \{1\} \\
    LF(\tuple{u,v}) &= \tuple{u,v} \\
    LF(P_1 + P_2) &= LF(P_1) \cup LF(P_2) \cup \{P_1 + P_2\} \\
    LF(P_1 \cdot P_2) &= LF(P_1) \cup LF(P_2) \cup \{P_1 \cdot
                        P_2\} \\
    LF(P^*) &= \{P^*\} 
  \end{align*}

  Note that $P \in LF(P)$ for all $P$.
\end{definition}

\begin{definition}
  % todo: find a better name
  A path expression $P$ is \emph{well-formed} if for all
  $P_1 \cdot P_2 \in SubExp(P)$,
  $LF(P_1) \cap LF(P_2) = \emptyset$.
\end{definition}

\begin{algorithm}
  \caption{Path Expression DAG to Transition Formula}
  \label{alg:path-exp-to-tf}

  \SetAlgoLined
  \SetAlgoNoEnd
  
  \KwIn{A control flow graph $G=\tuple{V, E, r, F}$ and a
    non-empty, well-formed path expression $P$ whose distinct
    subexpressions $\{P_1,\dots,P_n\}$ are labeled so that
    $P_i \in SubExp(P_j) \implies i < j$.}
  
  \KwOut{A formula $F$ such that
    $Sat(\tf{P}) \iff Sat(F)$}

  \ForEach{$i = 1$ \KwTo $n$}{
    \Switch{$P_i$}{
      \lCase{1}{
        $F_i \gets X_i = {X_i}'$
      }
      
      \lCase{$e \in E$}{
        $F_i \gets F_e(X_i, {X_i}')$
      }

      \lCase{$P_j + P_k$}{
        $F_i \gets (F_j \wedge X_i = X_j \wedge {X_i}' = {X_j}')
        \vee (F_k \wedge X_i = X_k \wedge {X_i}' = {X_k}')$
      }

      \lCase{$P_j \cdot P_k$}{
        $F_i \gets F_j \wedge F_k \wedge X_i = X_j \wedge {X_i}' =
        {X_k}' \wedge {X_j}' = X_k)$
      }

      \Case{${P_j}^*$}{

        $F^* \gets star(F_j, \tuple{X_j, {X_j}'})$ \;

        $F_i \gets F^*[X_j \mapsto X_i, {X_j}' \mapsto {X_i}']$ \;
        
      }
    }
  }

  $F \gets F_n$ \;

  \Return $F$ \;

\end{algorithm}

\begin{lemma}
  \label{pe:freevars}
  For all $1 \le j \le n$, the only free variables in $F_j$ are
  of the form $X_k$ or ${X_k}'$ for some $1 \le k \le j$ such
  that $P_k \in LF(P_j)$.
\end{lemma}

\begin{proof}
  We proceed by induction on the iteration $i$ of the for loop
  and assume the claim holds for each $j < i$. This is obviously
  the case before the first iteration. And, for any $i$ and any
  $j < i$, the claim still holds after the $i$th iteration since
  $F_j$ is not modified. For $j = i$, if $P_i = 1$,
  $P_i = e \in E$, or $P_i = {P_k}^*$ for some $k$, then the only
  free variables in $F_i$ are $X_i$, and ${X_i}'$, so we are
  done. Otherwise, $P_i = P_j + P_k$ or $P_i = P_j \cdot P_k$ for
  some $j$ and $k$, and the free variables in $F_i$ are exactly
  those in $F_j$ or $F_k$ or $X_i, X_j, X_k, {X_i}', {X_j}'$ or
  ${X_k}'$. By the induction hypothesis and the well-formedness
  of $P$, the claim holds for $i$.
\end{proof}

Let $V_j = \bigcup_{P_k \in LF(P_j)} X_k \cup {X_k}'$. Note that
by the above Lemma, if $P_j \in LF(P_i)$, then
$V_j \subseteq V_i$. As an abuse of notation, we write
$\exists V_j$ to denote
$\exists X_{k_1}, \dots, X_{k_l}, {X_{k_1}}', \dots, {X_{k_l}}'$,
where $LF(P_j) = \{P_{k_1}, \dots, P_{k_l}\}$.

\begin{theorem}
  $Sat(\tf{P}) \iff Sat(F)$
\end{theorem}

\begin{proof}
  We proceed by induction on the iteration $i$ of the for loop,
  with the hypothesis that before the $i$th iteration, for all
  $j < i$,
  $\tf{P_j} \equiv \exists V_j. F_j \wedge X_j = X \wedge {X_j}'
  = X'$. Since $F \equiv F_n$, and since $X$ and $X'$ appear
  nowhere else in $F$ by Lemma \ref{pe:freevars}, this is
  sufficient to prove the theorem.

  Certainly this is true before the first iteration. Suppose this
  is true before the $i$th iteration of the loop. Clearly, the
  claim still holds for $j < i$ after this iteration. For
  $j = i$, let us consider each possible form of $P_i$
  separately:

  \begin{itemize}
  \item $P_i = 1$: We have that $\tf{1} \equiv X = X'$ and
    $F_i \equiv X_i = {X_i}')$.  So,
    $\exists X_i, {X_i}'. F_i \wedge X_i = X \wedge {X_i}' = X'
    \equiv X = X' \equiv \tf{P_i}$.

  \item $P_i = \tuple{u, v}$: We have that
    $\tf{\tuple{u, v}} = F_{\tuple{u, v}}(X, X')$ and
    $F_i \equiv F_{\tuple{u, v}}(X_i, {X_i}')$.  So,
    $\exists X_i, {X_i}'. F_i \wedge X_i = X \wedge {X_i}' = X'
    \equiv F_{\tuple{u, v}}(X, X') \equiv \tf{P_i}$.
    
  \item $P_i = P_j + P_k$:
    $\tf{P_j+P_k} \equiv \tf{P_j} \vee \tf{P_k}$. By the
    induction hypothesis,
    $\tf{P_i} \equiv (\exists V_j. F_j \wedge X_j = X \wedge
    {X_j}' = X') \vee (\exists V_k. F_k \wedge X_k = X \wedge
    {X_k}' = X')$.

    Since $V_i = V_j \cup V_k \cup X_i \cup {X_i}'$, we have
    $\tf{P_i} \equiv \exists V_i. (F_j \wedge X_j = X_i \wedge
    {X_j}' = {X_i}') \vee (F_k \wedge X_k = X_i \wedge {X_k}' =
    {X_i}') \wedge X_i = X \wedge {X_i}' = X'$, as desired.
    
  \item $P_i = P_j \cdot P_k$:
    $\tf{P_j \cdot P_k} \equiv \exists X''_1, X''_2. \tf{P_j}(X,
    X''_1) \wedge \tf{P_k}(X''_2, X') \wedge X''_1 = X''_2$. So,
    by the induction hypothesis,
    $\tf{P_i} \equiv \exists X''_1, X''_2. (\exists V_j. F_j
    \wedge X_j = X \wedge {X_j}' = X''_1) \wedge (\exists
    V_k. F_k \wedge X_k = X''_2 \wedge {X_k}' = X') \wedge X''_1
    = X''_2$.

    Suppose $V_j \cap V_k \ne \emptyset$. Then, there must be
    some $P_l \in LF(P_j) \cap LF(P_k)$. But, this contradicts
    the well-formedness of $P$, from which we know that
    $LF(P_j) \cap LF(P_k) = \emptyset$. So,
    $V_j \cap V_k = \emptyset$. By Lemma \ref{pe:freevars}, this
    means there are no free variables present in both $F_j$
    and $F_k$. Therefore,
    \begin{align*}
      \tf{P_i}
      &\equiv \exists X''_1, X''_2. \exists V_i. F_j \wedge
        X_j = X \wedge {X_j}' = X''_1 \wedge F_k \wedge
        X_k = X''_2 \wedge {X_k}' = X' \wedge X''_1 = X''_2 \\
      &\equiv \exists V_i. F_j \wedge X_j = X \wedge
        {X_j}' = X_k \wedge F_k \wedge {X_k}' = X' \\
      & \equiv \exists V_i. F_i \wedge X_i = X \wedge {X_i}' = X'
    \end{align*}

  \item $P_i = {P_j}^*$: We know by the induction hypothesis that
    $\tf{P_j} \equiv \exists V_j. F_j \wedge X_j = X \wedge
    {X_j}' = X'$.  So,
    $F^* = star(F_j, \tuple{X_j, {X_j}'}) \equiv \tf{P_i}(X_j,
    {X_j}')$. Therefore,
    $F = F^*[X_j \mapsto X_i, {X_j}' \mapsto {X_i}'] \equiv
    \tf{P_i}(X_i, {X_i}')$. So,
    $\exists X_i, {X_i}'. F_i \wedge X_i = X \wedge {X_i}' = X'
    \equiv \tf{P_i}$.
  \end{itemize}
\end{proof}

\begin{algorithm}
  \caption{Modified Gaussian Elimination}
  \label{alg:gauss}

  \SetAlgoLined
  % \SetAlgoNoEnd
  
  \KwIn{A control flow graph $G=\tuple{V, E, r, F}$ with vertices
    labeled $1, \dots, n-1, n = r$}
  
  \KwOut{A function $pe: V \to PathExp(G)$ such that for all
    $v \in V$, $pe(v)$ is a well-formed path expression
    recognizing exactly $Paths_G(r, v)$}

  \tcc{$\si{P(u, v)} \subseteq Paths_G(u, v)$ for all
    $u, v \in V$}

  $P \gets \lambda \tuple{u, v}. 0$ \;

  \lForEach{$\tuple{u, v} \in E$}{
    $P(u, v) \gets P(u, v) \cdot \tuple{u, v}$
  }

  \For{$v = 1$ \KwTo $n-1$}{
    \For{$u = v + 1$ \KwTo $n$}{
      
      $P(u, v) \gets P(u, v) \cdot P(v, v)^*$ \;

      $P(u, u) \gets P(u, w) + (P(u, v) \cdot P(v, w))^*$ \;
      
      \For{$w = 1$ \KwTo $n$, $w \not\in \{u, v\}$}{
        $P(u, w) \gets P(u, w) + P(u, v) \cdot P(v, w)$ \;
      }
    }
  }

  $pe \gets \lambda v. P(r, v)$ \;

  \Return $pe$ \;

\end{algorithm}

\begin{lemma}
  After the $v$th iteration of the outer loop, for all
  $u, w \in V$, the following hold:
  \begin{enumerate}
  \item If $w \ne u$, then $P(u,w)$ is a well-formed path
    expression recognizing exactly the paths from $u$ to $w$ in
    $G$ whose intermediate vertices are smaller than
    $\min \{u, v+1\}$
  \item If $w = u$, then $P(u,w)=P(u,u)$ is a well-formed path
    expression recognizing exactly the cycles starting and ending
    at $u$ whose intermediate vertices are smaller than
    $\min \{u+1, v+1\}$
  \end{enumerate}
\end{lemma}

\begin{proof}
  We proceed by induction on $v$, and assume that before the
  $v$th iteration, (1) holds for $u < v$ and (2) holds for
  $u \ge v$ and intermediate vertices smaller than
  $\min \{u, v\}$. This is clearly true before the first
  iteration since all path expressions are either empty or
  edges. Consider any $u < v$; by the structure of the for loops,
  $P(u, w)$ for all $w$ is unchanged during this iteration, so
  (1) still holds at the end of this iteration.  Similarly, for
  all $w$, $P(v, w)$ is left unchanged, and we know the
  intermediate vertices in the paths it recognizes are smaller
  than $\min \{v, v\} = v$; ergo, (1) holds for such $w$ at the
  end of this iteration.

  Therefore, it only remains to prove (2) for any $u > v$. Note
  that $\min\{u, v\} = v$ and $\min\{u, v+1\} = v+1$. First, the
  only assignment to $P(u, v)$ this iteration is to concatenate
  $P(v, v)^*$ to it. $P(v, v)^*$ is well-formed with
  $LF(P(v, v)^*) = \{P(v, v)^*\}$, and
  $P(v, v)^* \not\in LF(P(u, v))$ before this iteration by the
  induction hypotheses. So, $P(u, v)$ is
  well-formed. Furthermore, consider any path from $u$ to $v$
  whose intermediate vertices are smaller than $v+1$. It must be
  the concatenation of a non-empty path $p_0$ from $u$ to $v$
  whose intermediate vertices are smaller than $v$ and $k$ paths
  $p_1, \dots, p_k$ from $v$ to $v$ whose intermediate vertices
  are smaller than $v$. By the induction hypothesis, $p_0$ is
  recognized by the previous $P(u, v)$, and each $p_i$ for
  $1 \le i \le k$ is recognized by $P(v, v)$. Thus, after this
  iteration $P(u, v)$ recognizes exactly the paths from $u$ to
  $v$ in $G$ whose intermediate vertices are smaller than $v+1$.

  Consider any $w \ne v$ and any path from $u$ to $w$ whose
  intermediate vertices are smaller than $v+1$. Either it is a
  path whose intermediate vertices are smaller than $v$, or it is
  the concatenation of a non-empty path $p_1$ from $u$ to $v$
  whose intermediate vertices are smaller than $v+1$ and a
  non-empty path $p_2$ from $v$ to $w$ whose intermediate
  vertices are smaller than $v$. Thus, $P(u, w) + P(u, v) \cdot
  P(v, w)$ recognizes exactly these paths by the induction
  hypothesis and our previous analysis of $P(u, v)$. Furthermore,
  we know that $P(u, w)$, $P(u, v)$, and $P(v, w)$ at this point
  are well-formed. Suppose for the sake of contradiction that
  there is some $P \in LF(P(u, v)) \cap LF(P(v, w))$.

  Note: it turns out this can be the case!
\end{proof}
