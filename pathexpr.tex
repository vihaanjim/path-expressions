\begin{definition}
  The set of \emph{regular expressions} over $\Sigma$, written
  $RegExp(\Sigma)$, is the smallest set such that
  $0, 1 \in RexExp(\Sigma)$, $\Sigma \subseteq RegExp(\Sigma)$,
  and $(R_1 + R_2), (R_1 \cdot R_2), R^* \in RegExp(\Sigma)$ for
  all $R, R_1, R_2 \in RegExp(\Sigma)$.

  $SubExp(R)$ for $R \in RegExp(\Sigma)$ is the set of
  subexpressions of $R$, including $R$ itself.
\end{definition}

\begin{definition}
  % taken more or less verbatim from Algebraic Program Analysis +
  % Fast Algorithms for Solving Path Problems
  The \emph{standard interpretation} is the function
  $\si{\cdot}: RegExp(\Sigma) \to \textbf{L}$, where $\textbf{L}$
  is the set of regular languages, such that
  $\si{0} = \emptyset$, $\si{1} = \{\epsilon\}$ where $\epsilon$
  is the empty string, $\si{a} = \{a\}$ for all $a \in \Sigma$,
  and
  \begin{align*}
  \si{R_1 + R_2} &= \si{R_1} \cup \si{R_2} \\
  \si{R_1 \cdot R_2} &= \{x_1x_2| x_1 \in \si{R_1}, x_2 \in
                       \si{R_2}\} \\
  \si{R^*} &= \{x_1x_2\dots x_n| x_1,x_2,\dots,x_n \in \si{R}\}
  \end{align*}
  for all $R, R_1, R_2 \in RegExp(\Sigma)$.
\end{definition}

\begin{definition}
  Given a function $wt: \Sigma \to \mathbb{Z}$, the \emph{length}
  of a regular expression is given by
  $L: RegExp(\Sigma) \to \mathbb{Z} \cup \{-\infty\}$ where
  $L(0) = -\infty$, $L(1) = L(R^*) = 1$, $L(a) = wt(a)$ for all
  $a \in \Sigma$, $L(R_1 + R_2) = \max\{L(R_1), L(R_2)\}$, and
  $L(R_1 \cdot R_2) = L(R_1) + L(R_2)$.
\end{definition}

Let $X$ and $X'$ be sets of variables such that for each
$x\in X$, there is a corresponding $x' \in X'$. As an abuse of
notation, we use $X = X' \equiv \bigwedge_{x \in X} x =
x'$. $Vars(F)$ for a formula $F$ is the set of free variables in
$F$. $Formula(X, X')$ denotes the set of formulas such that
$Vars(F) \subseteq X \cup X'$. 

\begin{definition}
  A \emph{control flow graph} $G = \tuple{V, E, r, F}$ is a graph
  with vertices $V$, edges $E$, a root vertex $r$ of indegree 0
  from which every vertex $v \in V$ is reachable, and a function
  $F: E \to Formula(X, X')$ for some $X, X'$. We use
  $F_{\tuple{u,v}}(Y, Y')$ to denote
  $F(\tuple{u,v})[X \mapsto Y, X' \mapsto Y']$. We let
  $Paths_G(u,v)$ be the set of all (possibly non-simple) paths
  from $u$ to $v$ in $G$.
\end{definition}

\begin{definition}
  A \emph{path expression} $P$ of type $\tuple{u, v}$ on
  $G = \tuple{V, E, r, F}$ is a regular expression over
  $E$ such that $\si{P} \subseteq Paths_G(u,v)$ for some
  $u,v \in V$. We denote the set of path expressions on $G$ by
  $PathExp(G)$.
\end{definition}

Suppose $star$ is a function that takes a formula $F$ and the
pair of variable vocabulairies $X$ and $X'$ that $F$ relates and
returns some overapproximation of the reflexive transitive
closure of $F$ over $X$ and $X'$ containing no other free
variables.

\begin{definition}
  The \emph{transition formula interpretation}
  is the function $\tf{\cdot}: PathExp(G) \to Formula(X, X')$
  where $\tf{0} \equiv false$, $\tf{1} \equiv X = X'$,
  $\tf{e} \equiv F_e(X, X')$ for all $e \in E$, and
  \begin{align*}
    \tf{P_a + P_b} &\equiv \tf{P_a} \lor \tf{P_b} \\
    \tf{P_a \cdot P_b} &\equiv \exists X''. \tf{P_a}(X, X'')
                         \land \tf{P_b}(X'', X') \\
    \tf{P^*} &\equiv star(\tf{P}, \tuple{X, X'}) 
  \end{align*}
  for all $P, P_a, P_b \in PathExp(G)$. 

  We use $\tf{P}(Y, Y')$ to denote
  $\tf{P}[X \mapsto Y, X' \mapsto Y']$.
\end{definition}

\begin{definition}
  Suppose there is an infinite set of variable vocabularies
  $\{X_0, X_1, X_2, \dots\}$, and suppose that $L$ is taken with
  respect to the function $wt: E \to \mathbb{Z}$ that maps every
  edge to 1. Then, define $I$ mapping path expressions and
  integer lower and upper bounds to formulas where
  $I(P, l, u) \equiv false$ if $L(P) = -\infty$, and otherwise,
  $I(1, l, u) \equiv X_l = X_u$,
  $I(e, l, u) \equiv F_e(X_l, X_u)$ for all $e \in E$, and
  \begin{align*}
    I(P_a + P_b, l, u)
    &\equiv I(P_a, l, u) \lor I(P_b, l, u) \\
    I(P_a \cdot P_b, l, u)
    &\equiv I(P_a, l, l + L(P_a)) \land I(P_b, l + L(P_a), u) \\
    I(P^*, l, u)
    &\equiv star(I(P, l+1, l+1+L(P)), \tuple{X_{l+1},
      X_{l+1+L(P)}})[X_{l+1} \mapsto X_l, X_{l+1+L(P)} \mapsto X_u]
  \end{align*}
  for all $P, P_a, P_b \in PathExp(G)$. 
\end{definition}

\begin{lemma}
  \label{pe:freevars}
  For all $P \in PathExp(G)$, for all $l \ge 0$ and
  $u \ge l + L(P)$,
  $Vars(I(P, l, u)) \subseteq \bigcup_{i = l}^u X_i$.
\end{lemma}

\begin{proof}
  The claim holds trivially when $L(P) = -\infty$, $P = 1$,
  $P = P'^*$, or $P = \tuple{v, w}$. Suppose the claim holds for
  $P_a$ and $P_b$ and that $L(P_a + P_b) \ge 0$. Take any
  $0 \le l \le l + L(P_a + P_b) \le u$. Since
  $L(P_a + P_b) = \max\{L(P_a), L(P_b)\}$, clearly
  $u \ge l + L(P_a)$ and $u \ge l + L(P_b)$. Then,
  $Vars(I(P_a + P_b, l, u)) \subseteq Vars(I(P_a, l, u)) \cup
  Vars(I(P_b, l, u)) = \bigcup_{i = l}^u X_i$. Similarly, suppose
  $L(P_a \cdot P_b) \ge 0$. Then, $L(P_a), L(P_b) \ge 0$. Take
  $0 \le l \le l + L(P_a \cdot P_b) = 1 + L(P_a) + L(P_b) \le u$.
  Clearly, $u \ge (l + L(P_a)) + P_b$. So,
  $Vars(I(P_a \cdot P_b, l, u)) \subseteq Vars(I(P_a, l, l +
  L(P_a))) \cup Vars(I(P_b, l + L(P_a), u)) = (\bigcup_{i = l}^{l
    + L(P_a)} X_i) \cup (\bigcup_{i = l + L(P_a)}^u X_i) =
  \bigcup_{i = l}^u X_i$. By induction, the claim must hold for
  all $P$.
\end{proof}

\begin{theorem}
  For all $P \in PathExp(G)$,
  $Sat(\tf{P}) \iff Sat(I(P, 0, L(P)))$.
\end{theorem}

\begin{proof}
  Let $V_{i,j} = \bigcup_{k=i}^j X_k$. As an abuse of notation,
  let $\exists S$ for a set $S = \{x_1, \dots, x_m\}$ denote
  $\exists x_1, \dots, x_m$. Then, we proceed by structural
  induction on $P$ with the hypothesis that for all $l \ge 0$ and
  $u \ge l + L(P)$,
  $\tf{P} \equiv \exists V_{l,u}. I(P, l, u) \land X = X_l \land
  X' = X_r$. By Lemma \ref{pe:freevars}, $X$ and $X'$ are not
  free in $I(P, l, u)$, so this is sufficient to prove the claim.
  If $L(P) = -\infty$, $P = 1$, or $P = \tuple{v, w}$, then the
  claim holds trivially.

  Suppose the claim holds for $P_a$ and $P_b$ and suppose
  $L(P_a + P_b) \ge 0$. Take arbitrary
  $0 \le l \le l + L(P_a + P_b) \le u$. Then,
  \begin{align*}
    \tf{P_a + P_b}
    &\equiv \tf{P_a} \lor \tf{P_b} \\
    &\equiv
      (\exists V_{l,u}. I(P_a, l, u) \land X = X_l \land X' = X_r)
      \lor (\exists V_{l,u}. I(P_b, l, u) \land X = X_l \land X' =
      X_r) \\
    &\equiv
      \exists V_{l,u}. I(P_a+P_b,l,u) \land X = X_l \land X' = X_r
  \end{align*}

  Similarly, suppose $L(P_a \cdot P_b) \ge 0$ and take arbitrary
  $0 \le l \le l + L(P_a \cdot P_b) \le u$. By Lemma
  \ref{pe:freevars}, we know that
  $Vars(I(P_a, l, l + L(P_a))) \subseteq V_{l, l + L(P_a)}$ and
  $Vars(I(P_b, l + L(P_a), u)) \subseteq V_{l + L(P_a), u}$.  By
  definition,
  $V_{l, u} = V_{l, l + L(P_a)} \cup V_{l + L(P_a), u}$ and
  $V_{l, l + L(P_a)} \cap V_{l + L(P_a), u} = X_{l +
    L(P_a)}$. Furthermore, $L(P_a), L(P_b) \ge 0$ by definition,
  meaning that $l + L(P_a) \ge 0$ and
  $u \ge (l + L(P_a)) + L(P_b)$. So,
  \begin{align*}
    \tf{P_a \cdot P_b}
    &\equiv \exists X''.
      \tf{P_a}(X, X'') \land \tf{P_b}(X'', X') \\
    &\equiv \exists X''.
      (\exists V_{l, l + L(P_a)}. I(P_a, l, l + L(P_a)) \land
      X = l \land X'' = X_{l + L(P_a)}) \\
    & \qquad \land (\exists V_{l + L(P_a), u}.
      I(P_b, l + L(P_a), u) \land
      X'' = X_{l + L(P_a)} \land X' = u) \\
    &\equiv \exists X''.
      \exists V_{l, u}.
      I(P_a, l, l + L(P_a)) \land I(P_b, l + L(P_a), u) \land
      X = l \land X'' = X_{l + L(P_a)} \land X' = u \\
    &\equiv \exists V_{l, u}.
      I(P_a \cdot P_b, l, u) \land X = l \land X' = u
  \end{align*}

  Finally, suppose the claim holds for $P$ and take arbitrary
  $0 \le l \le l + 1 \le u$. By the induction hypothesis,
  $\tf{P} \equiv \exists V_{l+1, l+1+L(P)}. I(P, l+1, l+1+L(P))
  \land X = X_{l+1} \land X' = X_{l+1+L(P)}$. So,
  $\tf{P^*}(X_{l+1}, X_{l+1+L(P)}) \equiv star(I(P, l+1, l+1+L(P)),
  \tuple{X_{l+1}, X_{l+1+L(P)}})$. Therefore,
  $\tf{P^*}(X_l, X_u) \equiv I(P^*, l, u)$, and the claim holds.
  
\end{proof}

\begin{definition}
  In a control flow graph $G = \tuple{V, E, r, F}$, $v\in V$ is
  said to \emph{dominate} $u \in V$ if every path from $r$ to $u$
  in $G$ goes through $v$. If $u \ne v$, then $v$ is said to
  \emph{strictly dominate} $u$. If $v$ strictly dominates $u$ and
  is dominated by every vertex that strictly dominates $u$, then
  $v$ is the \emph{immediate dominator} of $u$, denoted
  $v = idom(u)$. 
\end{definition}

\begin{lemma}
  \label{dominator:trans}
  If $x$ dominates $y$ and $y$ dominates $z$, then every path
  from $x$ to $z$ must go through $y$.
\end{lemma}

\begin{proof}
  If $x = y$ or $y = z$, then the claim holds trivially.
  Otherwise, let $p_1$ be a simple path from $r$ to $x$ and let
  $p_2$ be an arbitrary path from $x$ to $z$. By definition, the
  concatenation of $p_1$ and $p_2$ must go throught $y$. Suppose
  $p_1$ goes through $y$. Then, there is some prefix $p_1'$ of
  $p_1$ that is a path from $r$ to $y$. But, by definition,
  $p_1'$ must go through $r$, which contradicts the simplicity of
  $p_1$. Therefore, $p_2$ must go through $y$, as desired.
\end{proof}

\begin{definition}
  A control flow graph $G = \tuple{V, E, r, F}$ is
  \emph{reducible} if every cycle in $G$ contains an edge
  $\tuple{u, v}$ where $v$ dominates $u$. (Intuitively, every
  cycle has a natural entrypoint.)
\end{definition}

\begin{lemma}
  \label{reducible:order}
  If $G$ is reducible, then there is an ordering of the vertices
  such that for all $u, v\in V$, if there is some $u'$ that
  dominates $u$ but does not strictly dominate $v$ from which $v$
  is reachable, then $u \le v$.
\end{lemma}

\begin{algorithm}[H]
  \caption{Tarjan's Algorithm (on Reducible Flow Graphs)}
  \label{alg:tarjan}

  \SetAlgoLined
  % \SetAlgoNoEnd
  
  \KwIn{A reducible control flow graph $G=\tuple{V, E, r, F}$
    with vertices labeled $1,\dots, n = r$ as detailed in Lemma
    \ref{reducible:order}}
  
  \KwOut{A function $pe: V \to PathExp(G)$ such that for all
    $v \in V$, $pe(v)$ is a well-formed path expression
    recognizing exactly $Paths_G(r, v)$}

  Initialize compressed weighted forest on $V$ \;

  \For{$v = 1$ \KwTo $n-1$}{
    $cycle, noncycle \gets 0$ \;

    \ForEach{$v' \in pred(v)$}{
      \lIf{$v$ dominates $v'$}{
        $cycle \gets cycle + eval(v') \cdot \tuple{v', v}$
      }
      \lElse{
        $noncycle \gets noncycle + eval(v') \cdot \tuple{v', v}$
      }
    }

    $link(v, noncycle \cdot cycle^*, idom(v))$ \;
  }

  \Return $\lambda v. eval(v)$ \;

\end{algorithm}

Note that constant time simplication procedures applied after
each concatenation or union can easily remove all 0's and 1's
from the path expressions --- we assume that this is the case for
the rest of the analysis of Tarjan's algorithm.

\begin{lemma}
  \label{tarjan:correct}
  Before the $v$th iteration, for all $w < v$,
  $\si{eval(w)} = Paths_G(w', w) \cap E|_{w'}^*$, where $w'$ is
  the smallest vertex at least $v$ that dominates $w$.
\end{lemma}

\begin{proof}
  We proceed by induction on $v$. The claim holds vacuously
  before the first iteration. Suppose it holds before the $v$th
  iteration. $eval(w)$ is unchanged for all $w < v$ not dominated
  by $v$, and the smallest dominator stays the same as well, so
  the claim still holds for such $w$. Consider any $w$ dominated
  by $v$. By the ordering, $w \le v$, and the smallest dominator
  of $w$ that is at least $v$ is clearly $v$. By the induction
  hypothesis, before this iteration,
  $\si{eval(w)} = Paths_G(v, w) \cap E|_v^*$. Furthermore, the
  smallest dominator of $w$ strictly greater than $v$ is
  $idom(v)$, and every path from $idom(v)$ to $w$ must pass
  through $v$ by Lemma \ref{dominator:trans}. Thus, it is
  sufficient to show that
  $\si{noncycle \cdot cycle^*} = Paths_G(idom(v), v) \cap
  E|_{idom(v)}^*$. The fact that the left hand side is a subset
  of the right hand side follows trivially from the induction
  hypothesis, so we focus on the other direction.

  Consider any such path $p$. It can be chopped into paths
  $p_0, p_1,\dots,p_m$ where $p_0$ ends at $v$ and every $p_i$
  for $1 \le i \le m$ starts and ends at $v$, and no $p_i$ for
  $0 \le i \le m$ contains $v$ as an intermediate vertex. $p_0$
  can be further expressed as the concatenation of some path
  $p_0'$ and an edge $\tuple{v', v}$ for some $v'$ dominated by
  $idom(v)$. By Lemma \ref{dominator:trans}, $v$ does not
  dominate by $v'$ since $v$ does not appear in $p_0$. And, $v'$
  must be dominated by every vertex that strictly dominates $v$
  by the definition of $idom$. So, either $v' = idom(v)$, or $v'$
  is a vertex that does not dominate $v$ but from which $v$ is
  reachable, i.e. $v' \le v$. And, suppose there is some $v''$
  which dominates $v'$ but not $v$. Then, $v$ is reachable from
  $v''$, so $v'' \le v$. Therefore, the smallest vertex at least
  $v$ that dominates $v''$ is $idom(v)$. By the induction
  hypothesis,
  $\si{eval(v')} = Paths_G(idom(v), v') \cap E_{idom(v)}^*$. So,
  $noncycle$ recognizes $p_0$.

  Now, take any of the $p_i$. It can be further expressed as the
  concatenation of some path $p_i'$ and an edge $\tuple{v', v}$,
  for some $v'$ strictly dominated by $idom(v)$. By the
  definition of $idom$, $v'$ does not strictly dominate $v$. So,
  since $v'$ dominates itself, $v' < v$. But, $v'$ is reachable
  from $v$, meaning that $v$ must strictly dominate $v'$ because
  otherwise we get that $v \le v'$. Therefore, by the induction
  hypothesis, $p_i'$ is recognized by $eval(v')$ before this
  iteration. So, $cycle$ recognizes $p_i$ for all
  $1 \le i \le m$. Therefore, the claim holds after this
  iteration.
\end{proof}

\begin{lemma}
  Before the $v$th iteration, there exist
  $l, u : S \to \mathbb{Z} \cup \{-\infty\}$ where
  $S = \bigcup_{w = 1}^{v-1} SubExp(eval(w))$ such that the
  following hold for all $w < v$:
  \begin{enumerate}
  \item $l(eval(w)) = 0$ and $u(eval(w)) = L(eval(w))$
  \item For all $P_a + P_b \in SubExp(eval(w))$,
    $l(P_a) = l(P_b) = l(P_a + P_b)$ and
    $u(P_a) = u(P_b) = u(P_a + P_b)$.
  \item For all $P_a \cdot P_b \in SubExp(eval(w))$,
    $l(P_a) = l(P_a \cdot P_b)$, $u(P_a) = l(P_b) = l(P_a \cdot
    P_b) + L(P_a)$, and $u(P_a) = u(P_a \cdot P_b)$.
  \item For all $P^* \in SubExp(eval(w))$,
    $l(P) = l(P^*) + 1$ and $u(P) = u(P^*) + 1 + L(P)$.
  \end{enumerate}
\end{lemma}

\begin{proof}
  (quick and dirty!)
  We proceed by induction on $v$. The claim obviously holds prior
  to the first iteration. Suppose it holds before the $v$th
  iteration for some particular $l$ and $u$.  Let
  $W_1 = \{w<v|idom(v) \text{ dominates } w\}$ and
  $W_2 = \{w<v|idom(v) \text{ doesn't dominate } w\}$. Then, let
  $S_1 = \bigcup_{W_1} SubExp(eval(w))$ and
  $S_2 = \bigcup_{W_2} SubExp(eval(w))$. Clearly,
  $S = S_1 \cup S_2$. By the transitivity of the domination
  relation and Lemma \ref{tarjan:correct}, the only edges in
  $S_1$ are members of $E|_{idom(v)}$. Consider any $w \in W_2$
  and let $w'$ be the smallest dominator of $w$ at least $v$. If
  $w'$ does not dominate $idom(v)$, then
  $E|_{idom(v)} \cap E_{w'} = \emptyset$, so
  $S_1 \cap SubExp(eval(w)) = \emptyset$ by Lemma
  \ref{tarjan:correct} since $0, 1 \not \in S$. Suppose that $w'$
  does dominate $idom(v)$. By Lemma \ref{tarjan:correct},
  $\si{eval(w)} = Paths_G(w', w) \cap E|_{w'}^*$. But, since
  $idom(v)$ does not dominate $w$ and $idom(v) \ge v \ge w$,
  $Paths_G(idom(v), w) \cap E|_{w'}^* = \emptyset$. Therefore,
  $S_1 \cap SubExp(eval(w)) = \emptyset$. So,
  $S_1 \cap S_2 = \emptyset$.

  Let $eval(\cdot)$ denote the state of the compressed weighted
  forest before this iteration and $eval'(\cdot)$ denote its
  state after. Then, note that the elements of
  $(\bigcup_{w = 1}^v SubExp(eval'(w))) \setminus S$ are
  $\tuple{v', v}$ and the concatenations
  $eval(v') \cdot \tuple{v', v}$ for $v'\in pred(v)$, which are
  all dominated by $idom(v)$, the successive unions to $cycle$
  and $noncycle$, $cycle^*$, $noncycle \cdot cycle^*$, and the
  new $eval'(w) = noncycle \cdot cycle^* \cdot eval(w)$ for all
  $w$ dominated by $v$. Since the $\tuple{v', v}$ are new and
  distinct, if we let $l(noncycle) = 0$ and
  $u(noncycle) = L(noncycle)$, the claim holds if the values we
  assign $eval(v')$ in $cycle$ align with the values we assign it
  in $eval'(v')$ for all $v'$ dominated by $v$. But, this comes
  directly from (4) since we know that
  $L(noncycle \cdot cycle^*) = L(noncycle) + 1$. Therefore, the
  claim holds after iteration $v$. 
\end{proof}
