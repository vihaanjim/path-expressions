A few minor style notes:
\begin{itemize}
\item Avoid multi-character strings in math mode -- escape with mathit
  (certain combinations of characters look strange).  (Compare
  $Math\ Mode$ vs $\mathit{Math\ Mode}$)
\item I ususally don't use definition blocks unless it's a definition
  that I'm introducing in the paper.  The background section of paper
  often contains dozens of defintions so it can look cluttered.  My
  personal style is to use \textbf{boldface} for inline definitions so
  that readers can find them more easily, and to use \emph{italics}
  for forward-reference to terms that remain to be defined (signals to
  the reader: ``you're not exected to know what this is yet'').  I
  also like to use $\defeq$ to distinguish definitional equality (x
  $\defeq$ y means that x is defined to be y) vs judgement of equality
  (5 = 2+3 by computation, not definition)
\item Prefer $\ell$ to $l$
\item Good practice to use macros whenever you introduce notation.  I
  added some examples above.  Sometimes we want to switch notation,
  and easier if it's a macro.
\item Prefer $=$ to $\equiv$ when formulas are syntactically equal --
  reduces the burden on the reader because it signals a trivial step.
\end{itemize}

\begin{definition}
  Given a function $wt: \Sigma \to \mathbb{Z}$, the \defn{length}
  of a regular expression is given by
  $L: \regexp{\Sigma} \to \mathbb{Z} \cup \{-\infty\}$ where
  $L(0) \defeq -\infty$, $L(1) \defeq 0$, $L(R^*) \defeq 1$,
  $L(a) \defeq wt(a)$ for all $a \in \Sigma$,
  $L(R_1 + R_2) \defeq \max\{L(R_1), L(R_2)\}$, and
  $L(R_1 \cdot R_2) \defeq L(R_1) + L(R_2)$.
\end{definition}

\begin{definition}
  Suppose there is an infinite set of variable vocabularies
  $\{X_0, X_1, X_2, \dots\}$, and suppose that $L$ is taken with
  respect to the function $wt: E \to \mathbb{Z}$ that maps every
  edge to 1. Then, define $F$ mapping path expressions and
  integer lower and upper bounds to formulas where
  $F(P, \ell, u) \defeq \false$ if $L(P) = -\infty$, and
  otherwise $F(1, \ell, u) \defeq X_\ell = X_u$,
  $F({\tuple{u, v}}, \ell, u) \defeq \mathscr{F}_{\tuple{u,
      v}}(X_\ell, X_u)$, and
  \begin{align*}
    F(P_1 + P_2, \ell, u)
    &\defeq F(P_1, \ell, u) \lor F(P_2, \ell, u) \\
    F(P_1 \cdot P_2, \ell, u)
    &\defeq F(P_1, \ell, \ell + L(P_1)) \land F(P_2, \ell + L(P_1), u) \\
    F(P^*, \ell, u)
    &\defeq \tc(F(P, \ell+1, \ell+1+L(P)), \tuple{X_{\ell+1},
      X_{\ell+1+L(P)}})[X_{\ell+1} \mapsto X_\ell, X_{\ell+1+L(P)} \mapsto X_u]
  \end{align*}
  for all $P, P_1, P_2 \in \pathexp{G}$. 
\end{definition}

Let $V_{i,j} = \bigcup_{k=i}^j X_k$. As an abuse of notation, let
$\exists S$ for a set $S = \{z_1, \dots, z_N\}$ denote

\begin{lemma}
  \label{pe:freevars}
  For all $P \in \pathexp{G}$, for all $\ell \ge 0$ and
  $u \ge \ell + L(P)$, $\fv{F(P, \ell, u)} \subseteq V_{\ell,u}$.
\end{lemma}

\begin{proof}
  The claim holds trivially when $L(P) = -\infty$, $P = 1$,
  $P = P'^*$, or $P = \tuple{v, w}$. Suppose the claim holds for
  $P_1$ and $P_2$ and that $L(P_1 + P_2) \ge 0$. Take any
  $0 \le l \le \ell + L(P_1 + P_2) \le u$. Since
  $L(P_1 + P_2) = \max\{L(P_1), L(P_2)\}$, clearly
  $u \ge \ell + L(P_1)$ and $u \ge \ell + L(P_2)$. Then,
  $\fv{F(P_1 + P_2, \ell, u)} \subseteq \fv{F(P_1, \ell, u)} \cup
  \fv{F(P_2, \ell, u)} = V_{\ell, u}$. Similarly, suppose
  $L(P_1 \cdot P_2) \ge 0$. Then, $L(P_1), L(P_2) \ge 0$. Take
  $0 \le l \le \ell + L(P_1 \cdot P_2) = 1 + L(P_1) + L(P_2) \le
  u$.  Clearly, $u \ge (\ell + L(P_1)) + P_2$. So,
  $\fv{F(P_1 \cdot P_2, \ell, u)} \subseteq \fv{F(P_1, \ell, \ell
    + L(P_1))} \cup \fv{F(P_2, \ell + L(P_1), u)} = V_{\ell, \ell
    + L(P_1)} \cup V_{\ell + L(P_1),u} = V_{\ell, u}$. By
  induction, the claim must hold for all $P$.
\end{proof}

\begin{theorem}
  For all $\ell \ge 0$ and $u \ge \ell + L(P)$,
  $\tf{P} \equiv \exists V_{\ell,u}. F(P, \ell, u) \land X =
  X_\ell \land X' = X_u$
\end{theorem}

\begin{proof}
  We proceed by structural induction on $P$. If $L(P) = -\infty$,
  then the claim holds trivially. Otherwise, suppose
  $L(P) \ge 0$, and take arbitrary
  $0 \le \ell \le \ell + L(P) \le u$. If $P = 1$ or
  $P = \tuple{v, w}$, then the claim also holds trivially.

  Suppose $P = P_1 + P_2$. By the induction hypothesis,
  \begin{align*}
    \tf{P_1 + P_2}
    &= \tf{P_1} \lor \tf{P_2} \\
    &\equiv
      (\exists V_{\ell,u}. F(P_1, \ell, u) \land X = X_\ell \land X' = X_r)
      \lor (\exists V_{\ell,u}. F(P_2, \ell, u) \land X = X_\ell \land X' =
      X_u) \\
    &\equiv
      \exists V_{\ell,u}. F(P_1+P_2,\ell,u) \land X = X_\ell \land X' = X_u
  \end{align*}

  Similarly, suppose $P = P_1 \cdot P_2$. By Lemma
  \ref{pe:freevars}, we know that
  $\fv{F(P_1, \ell, \ell + L(P_1))} \subseteq V_{\ell, \ell +
    L(P_1)}$ and
  $\fv{F(P_2, \ell + L(P_1), u)} \subseteq V_{\ell + L(P_1), u}$.
  By definition,
  $V_{\ell, u} = V_{\ell, \ell + L(P_1)} \cup V_{\ell + L(P_1),
    u}$ and
  $V_{\ell, \ell + L(P_1)} \cap V_{\ell + L(P_1), u} = X_{\ell +
    L(P_1)}$. Furthermore, $L(P_1), L(P_2) \ge 0$ since
  $L(P_1 \cdot P_2) \geq 0$. So, $\ell + L(P_1) \ge 0$ and
  $u \ge (\ell + L(P_1)) + L(P_2)$. By the induction hypothesis,
  \begin{align*}
    \tf{P_1 \cdot P_2}
    &= \exists X''.
      \tf{P_1}(X, X'') \land \tf{P_2}(X'', X') \\
    &\equiv \exists X''.
      (\exists V_{\ell, \ell + L(P_1)}. F(P_1, \ell, \ell + L(P_1)) \land
      X = l \land X'' = X_{\ell + L(P_1)}) \\
    & \qquad \land (\exists V_{\ell + L(P_1), u}.
      F(P_2, \ell + L(P_1), u) \land
      X'' = X_{\ell + L(P_1)} \land X' = u) \\
    &\equiv \exists X''.
      \exists V_{\ell, u}.
      F(P_1, \ell, \ell + L(P_1)) \land F(P_2, \ell + L(P_1), u) \land
      X = l \land X'' = X_{l + L(P_1)} \land X' = u \\
    &\equiv \exists V_{\ell, u}.
      F(P_1 \cdot P_2, \ell, u) \land X = l \land X' = u
  \end{align*}

  Finally, suppose the claim holds for $P$ and take arbitrary
  $0 \le l \le \ell + 1 \le u$. By the induction hypothesis,
  $\tf{P} \equiv \exists V_{\ell+1, \ell+1+L(P)}. F(P, \ell+1, \ell+1+L(P))
  \land X = X_{\ell+1} \land X' = X_{\ell+1+L(P)}$. So,
  $\tf{P^*}(X_{\ell+1}, X_{\ell+1+L(P)}) \equiv \tc(F(P, \ell+1, \ell+1+L(P)),
  \tuple{X_{\ell+1}, X_{\ell+1+L(P)}})$. Therefore,
  $\tf{P^*}(X_\ell, X_u) \equiv F(P^*, \ell, u)$, and the claim holds.

  \zak{The star case is non-trivial -- no reason that
    $star(F(X_0,X_`),\tuple{X_0,X_1})$ should neceesarily be

\section{Space-efficient encodings of reducible flow graphs}
\begin{lemma}
  \label{dom:trans}
  If $u \dominates v \dominates w$, then every path
  from $u$ to $w$ passes through $v$.
\end{lemma}
\begin{proof}
  If $u = v$ or $v = w$, then the claim holds trivially.
  Otherwise, let $p_1$ be a simple path from $r$ to $u$ and let
  $p_2$ be an arbitrary path from $u$ to $w$. By definition, the
  concatenation of $p_1$ and $p_2$ passes through $v$. Suppose
  $p_1$ passes through $v$. Then, there is some prefix $p_1'$ of
  $p_1$ that is a path from $r$ to $v$. But, by definition,
  $p_1'$ passes through $u$, which contradicts the simplicity of
  $p_1$. Therefore, $p_2$ passes through $v$, as desired.
\end{proof}

\begin{lemma}
  \label{dom:nostrict}
  If $u \notstrictlydominates v$, $v \notstrictlydominates u$,
  $u \notstrictlydominates w$, and $v \dominates w$, then every
  path from $u$ to $w$ passes through $v$.
\end{lemma}

\begin{proof}
  If $u = v$ or $v = w$, then the claim holds trivially. If
  $u = w$, then $v = w$, so the claim also holds
  trivially. Otherwise, consider a path $p_1$ from $r$ to $u$
  that doesn't pass through $v$. Then, take any path $p_2$ from
  $u$ to $w$. The concatenation of $p_1$ and $p_2$ is a path from
  $r$ to $w$, so it passes through $v$. So, $p_2$ passes through
  $v$.
\end{proof}

\begin{lemma}
  \label{dom:backedge}
  If $v \dominates u$, then there is a path from $v$ to $u$ that
  doesn't traverse any back edges.
\end{lemma}

\begin{proof}
  Take a simple path $p$ from $r$ to $u$. Suppose $p$ traverses a
  back edge $\tuple{w_1,w_2}$. Let $p'$ be the prefix of $p$ that
  comes immediately before this edge. Then, $p'$ must go through
  $w_2$, which contradicts the simplicity of $p$. Since $p$ goes
  through $v$ by definition, this completes the proof.
\end{proof}

\begin{lemma}
  \label{reducible:order}
  If $G$ is reducible, then there is an ordering of the vertices
  such that for all $u, v\in V$, $u \le v$ if there is a vertex
  $u'$ where $u \dominatedby u' \notstrictlydominates v$ and
  there is a path from $u'$ to $v$ that doesn't traverse any back
  edges.
\end{lemma}

\begin{proof}
  We prove the claim by showing that on reducible graphs, the
  above relation is a partial order.
  \begin{enumerate}
  \item Reflexivity: let $v \in V$ be arbitrary.
    $v \dominatedby v \notstrictlydominates v$, and the empty
    path from $v$ to $v$ does not traverse any back edges.
  \item Antisymmetry: suppose there are some $u, u', v, v' \in V$
    and paths $p_1$ and $p_2$ between $u'$ and $v$ and $v'$ and
    $u$, respectively, such that
    $u \dominatedby u' \notstrictlydominates v$,
    $v \dominatedby v' \notstrictlydominates u$, and $p_1$ and
    $p_2$ don't traverse any back edges. Since
    $u' \notstrictlydominates v$ and
    $v' \notstrictlydominates u$, neither $u'$ nor $v'$ can
    strictly dominate the other. By Lemma \ref{dom:nostrict},
    $p_1$ goes through $v'$, and $p_2$ goes through $u'$. But,
    this means there is a cycle starting and ending at $u'$ that
    goes through $v'$ and doesn't traverse a back edge. If the
    cycle is non-empty, then no vertex dominates its predecessor
    in the cycle. Since $G$ is reducible, this means the cycle
    must be empty, i.e. $u' = v'$. So, $u' \dominates v$ but
    $u' \notstrictlydominates v$ and $v' \dominates u$ but
    $v' \notstrictlydominates u$. Therefore, $u = v' = u' = v$.
  \item Transitivity: suppose there are some
    $u, u', v, v', w \in V$ and paths $p_1$ and $p_2$ between
    $u'$ and $v$ and $v'$ and $w$, respectively, such that
    $u \dominatedby u' \notstrictlydominates v$,
    $v \dominatedby v' \notstrictlydominates w$, and $p_1$ and
    $p_2$ don't traverse any back edges. Note that
    $u' \notstrictlydominates v'$. If $v' \dominates u'$, then we
    are done. Otherwise, $v' \notstrictlydominates u'$. By Lemma
    \ref{dom:nostrict}, $p_1$ passes through $v'$. But, we also
    have that $u' \notstrictlydominates w$, because otherwise
    $u' \dominates v'$. So, there is a path from
    $u' \dominates u$ to $w \notstrictlydominatedby u'$ through
    $v'$ that doesn't traverse any back edges.
  \end{enumerate}
\end{proof}

\begin{algorithm}[H]
  \caption{Tarjan's Path Expression Algorithm on Reducible Flow Graphs}
  \label{alg:tarjan}

  \SetAlgoLined
  % \SetAlgoNoEnd
  
  \KwIn{A reducible control flow graph
    $G=\tuple{V, E, r, \mathscr{F}}$ with vertices labeled
    $1,\dots, n = r$ as detailed in Lemma \ref{reducible:order}}
  
  \KwOut{A function $pe: V \to \pathexp{G}$ such that for all
    $v \in V$, $pe(v)$ is a well-formed path expression
    recognizing exactly $\Paths{G}{r}{v}$}

  Initialize a compressed weighted forest on $V$ \;

  \For{$v = 1$ \KwTo $n-1$}{
    $cycle, noncycle \gets 0$ \;

    \ForEach{$v' \in pred(v)$}{
      \lIf{$\find{v'} = v$}{
        $cycle \gets cycle + \eval{v'} \cdot \tuple{v', v}$
      }
      \lElse{
        $noncycle \gets noncycle + \eval{v'} \cdot \tuple{v', v}$
      }
    }

    $\link{v}{noncycle \cdot cycle^*}{idom(v)}$ \;
  }

  \Return $\lambda v. \eval{v}$ \;

\end{algorithm}

\begin{lemma}
  \label{tarjan:correct}
  Before the $v$th iteration, for all $w < v$, $\find(w) = D_v(w)$
  and $\si{\eval{w}} = \Paths{G}{D_v(w)}{w} \cap E|_{D_v(w)}^*$,
  where $D_v(w)$ is the smallest vertex at least $v$ that
  dominates $w$.
\end{lemma}

\begin{proof}
  We proceed by induction on $v$. The claim holds vacuously
  before the first iteration. Suppose it holds before the $v$th
  iteration. By the induction hypothesis, $\find(w) = v$ only if
  $v \dominates w$. So, for all $w \notdominatedby v$, $\find(w)$
  and $\eval{w}$ are unchanged, and $D_{v+1}(w) = D_v(w)$. So,
  the claim still holds for such $w$. Consider any
  $w \dominatedby v$. By Lemma \ref{dom:backedge}, $w \le
  v$. And, $D_v(w) = v$. By the induction hypothesis, before this
  iteration, $\si{\eval{w}} = \Paths{G}{v}{w} \cap
  E|_v^*$. Furthermore, $D_{v+1}(w) = idom(v)$, and every path
  from $idom(v)$ to $w$ must pass through $v$ by Lemma
  \ref{dom:trans}. Thus, it is sufficient to show that
  $\si{noncycle \cdot cycle^*} = \Paths{G}{idom(v)}{v} \cap
  E|_{idom(v)}^*$. That the left hand side is a subset of the
  right hand side follows trivially from the induction
  hypothesis, so we focus on the other direction.

  Consider any such path $p$. It can be chopped into paths
  $p_0, p_1,\dots,p_m$ where no $p_i$ for $0 \le i \le m$
  contains $v$ as an intermediate vertex. $p_0$ can be further
  expressed as the concatenation of a path $p_0'$ and an edge
  $\tuple{v', v}$ where $v' \dominatedby idom(v)$. By Lemma
  \ref{dom:trans}, since $p_0'$ doesn't go through $v$,
  $v \notdominates v'$. And, since $v' \dominatedby idom(v)$,
  $v'$ must be dominated by every vertex that strictly dominates
  $v$. So, either $v' = idom(v)$, or $v' \notstrictlydominates v$
  and $\tuple{v', v}$ is a path from $v'$ to $v$ that doesn't
  traverse a back edge, i.e.  $v' \le v$. Suppose there is some
  $v''$ such that $v' \dominatedby v'' \notdominates v$. By Lemma
  \ref{dom:backedge}, there is a path from $v''$ to $v'$ that
  does not traverse a back edge. So, $v'' < v$. Therefore, the
  smallest vertex at least $v$ that dominates $v'$ is
  $idom(v)$. By the induction hypothesis, $p_0'$ is recognized by
  $\eval{v'}$. So, $noncycle$ recognizes $p_0$.

  Now, take any of the $p_i$. By reducibility, some vertex in
  $p_i$ must dominate all the others. In particular, it must
  dominate $v$. But, $p_i$ does not pass through
  $idom(v)$. Therefore, $v$ must dominate all vertices in
  $p_i$. In particular, we can express $p_i$ as the concatenation
  of a path $p_i'$ and an edge $\tuple{v', v}$, where
  $v' \dominatedby v$. By the induction hypothesis, $p_i'$ is
  recognized by $\eval{v'}$ before this iteration. So, $cycle$
  recognizes $p_i$ for all $1 \le i \le m$. Therefore, the claim
  holds after this iteration.
\end{proof}

\begin{definition}

  $\type{P_0} = \tuple{\tuple{\src{P_0}, \sstar{P_0}},
    \tuple{\tgt{P_0}, \tstar{P_0}}}$ and
  $\starred{P_0} \equiv \sstar{P_0} \land \tstar{P_0}$.
\end{definition}

\begin{lemma}
  Suppose $\eval[v]{\cdot}$ is the state of the compressed
  weighted forest before the $v$th iteration. Let
  $S_v = \bigcup_{w < v} \subexp{\eval[v]{w}}$,
  $K_v^t = \max\{L(P)|P \in S_v, \type{P} = t\}$, and
  \begin{align*}
    \ell_{v,k}(P)
    &= \begin{cases}
      k+L(\eval[v]{\src{P}}) & \neg \starred{P} \\
      k+L(\eval[v]{\tgt{P}}) - 1 & \starred{P}
    \end{cases} \\
    u_{v,k}(P) &= \ell_{v,k}(P) + K_v^{\type{P}}
  \end{align*}
  for any $k \ge 0$. Then, for all $k \ge 0$, the following hold:

  \begin{enumerate}
  \item For all $P = P_1 + P_2 \in S_v$,
    $F(P, \ell_{v,k}(P), u_{v,k}(P)) \equiv F(P_1, \ell_{v,k}(P_1),
    u_{v,k}(P_1)) \lor F(P_2, \ell_{v,k}(P_2), u_{v,k}(P_2))$.
  \item For all $P = P_1 \cdot P_2 \in S_v$,
    $F(P, \ell_{v,k}(P), u_{v,k}(P)) \equiv F(P_1, \ell_{v,k}(P_1),
    u_{v,k}(P_1)) \land F(P_2, \ell_{v,k}(P_2), u_{v,k}(P_2))$
