\begin{definition}
  A \emph{regular expression over $\Sigma$} is an element of the
  set $RegExp(\Sigma)$, which is the smallest set satisfying the
  following properties:

  \begin{itemize}
  \item For all $a \in \Sigma$, $a \in RegExp(\Sigma)$.
  \item $0, 1 \in RegExp(\Sigma)$.
  \item For all $R_0 \in RegExp(\Sigma)$,
    ${R_0}^* \in RegExp(\Sigma)$.
  \item For all $R_1, R_2 \in RegExp(\Sigma)$,
    $(R_1 + R_2), (R_1 \cdot R_2) \in RegExp(\Sigma)$.
  \end{itemize}

  A regular expression is \emph{simple} if it is 0 or does not
  have 0 as a subexpression. 
\end{definition}

\begin{definition}
  % taken more or less verbatim from Algebraic Program Analysis +
  % Fast Algorithms for Solving Path Problems
  The \emph{standard interpretation} is the function
  $\si{\cdot}: RegExp(\Sigma) \to \textbf{L}$, where $\textbf{L}$
  is the set of regular languages, such that
  $\si{0} = \emptyset$, $\si{1} = \{\epsilon\}$ where $\epsilon$
  is the empty string, $\si{a} = \{a\}$ for all $a \in \Sigma$,
  and
  \begin{align*}
  \si{R_1 + R_2} &= \si{R_1} \cup \si{R_2} \\
  \si{R_1 \cdot R_2} &= \{x_1x_2| x_1 \in \si{R_1}, x_2 \in
                       \si{R_2}\} \\
  \si{R^*} &= \{x_1x_2\dots x_n| x_1,x_2,\dots,x_n \in \si{R}\}
  \end{align*}
  for all $R, R_1, R_2 \in RegExp(\Sigma)$.
\end{definition}

\begin{definition}
  A \emph{control flow graph} $G = \tuple{V, E, r, F}$ is a graph
  with vertices $V$, edges $E$, a root vertex $r$ of indegree 0
  from which every vertex $v \in V$ is reachable from $r$, and a
  function $F: E \to Formula(X, X')$ for
  $X, X' \in \mathbb{Z}^{Var}$. We use $F_{\tuple{u,v}}(Y, Y')$
  to denote $F(\tuple{u,v})[X \mapsto Y, X' \mapsto Y']$. We let
  $Paths_G(u,v)$ be the set of all (possibly non-simple) paths
  from $u$ to $v$ in $G$.

\end{definition}

\begin{definition}
  A \emph{path expression} $P$ of type $\tuple{u, v}$ on
  $G = \tuple{V, E, r, F}$ is a simple regular expression over
  $E$ such that $\si{P} \subseteq Paths_G(u,v)$ for some
  $u,v \in V$. We denote the set of path expressions on $G$ by
  $PathExp(G)$.

\end{definition}

Let $X$ and $X'$ be sets of variables such that for each
$x\in X$, there is a corresponding $x' \in X'$. As an abuse of
notation, we use $X = X' \equiv \bigwedge_{x \in X} x = x'$. Let
$star$ be a function that takes a formula and the pair of
variable vocabulairies that it relates and returns some
overapproximation of the reflexive transitive closure of the
formula over those vocabularies, containing no other free
variables.

\begin{definition}
  The \emph{transition formula interpretation}
  is the function $\tf{\cdot}: PathExp(G) \to Formula(X, X')$
  where $\tf{0} \equiv false$, $\tf{1} \equiv X = X'$,
  $\tf{e} \equiv F_e(X, X')$ for all $e \in E$, and
  \begin{align*}
    \tf{P_1 + P_2} &\equiv \tf{P_1} \vee \tf{P_2} \\
    \tf{P_1 \cdot P_2} &\equiv \exists X''. \tf{P_1}(X, X'')
                         \wedge \tf{P_2}(X'', X') \\
    \tf{P^*} &\equiv star(\tf{P}, \tuple{X, X'}) 
  \end{align*}
  for all $P, P_1, P_2 \in PathExp(G)$. 

  We use $\tf{P}(Y, Y')$ to denote
  $\tf{P}[X \mapsto Y, X' \mapsto Y']$.
\end{definition}

\begin{definition}
  % todo: find a better name
  A path expression $P$ is \emph{well-formed} if there exists a
  function $LF: SubExp(P) \to \mathscr{P}(SubExp(P))$ such that
  the following hold:
  \begin{itemize}
  \item If $1 \in SubExp(P)$, then $LF(1) = \{1\}$
  \item For all $e = \tuple{u,v} \in SubExp(P)$,
    $LF(e) = \{e\}$.
  \item For all $P_1 + P_2 \in SubExp(P)$,
    $LF(P_1+P_2) = LF(P_1) \cup LF(P_2) \cup \{P_1 + P_2\}$.
  \item For all $P_1 \cdot P_2 \in SubExp(P)$,
    $LF(P_1 \cdot P_2) = LF(P_1) \cup LF(P_2) \cup \{P_1 \cdot
    P_2\}$ where $LF(P_1) \cap LF(P_2) = \emptyset$.
  \item For all ${P_1}^* \in SubExp$,
    $LF({P_1}^*) = \{{P_1}^*\}$.
  \end{itemize}
\end{definition}

\begin{algorithm}
  \caption{Path Expression DAG to Transition Formula}
  \label{alg:path-exp-to-tf}

  \SetAlgoLined
  \SetAlgoNoEnd
  
  \KwIn{A control flow graph $G=\tuple{V, E, r, F}$ and a
    non-empty, well-formed path expression $P$ whose distinct
    subexpressions $\{P_1,\dots,P_n\}$ are labeled so that
    $P_i \in SubExp(P_j) \implies i < j$.}
  
  \KwOut{A formula $\Gamma$ such that
    $Sat(\tf{P}) \iff Sat(\Gamma \wedge b_n)$}

  $\Gamma \gets \lambda x. true$
  
  \ForEach{$i = 1$ \KwTo $n$}{
    \Switch{$P_i$}{
      \lCase{1}{
        $\Gamma_i \gets (b_i \implies X_i = {X_i}')$
      }
      
      \lCase{$e \in E$}{
        $\Gamma_i \gets (b_i \implies F_e(X_i, {X_i}'))$
      }

      \Case{$P_j + P_k$}{
        
        $\Gamma_i \gets \Gamma_j \wedge \Gamma_k \wedge (b_i
        \implies (b_j \wedge X_i = X_j \wedge {X_i}' = {X_j}')
        \vee (b_k \wedge X_i = X_k \wedge {X_i}' = {X_k}'))$ \;
        
      }

      \lCase{$P_j \cdot P_k$}{
        $\Gamma_i \gets \Gamma_j \wedge \Gamma_k \wedge (b_i
        \implies b_j \wedge b_k \wedge X_i = X_j \wedge {X_i}' =
        {X_k}' \wedge {X_j}' = X_k))$
      }

      \Case{${P_j}^*$}{

        $F \gets star(\Gamma_j \wedge b_j, \tuple{X_j, {X_j}'})$
        \;

        $F' \gets F[X_j \mapsto X_i, {X_j}' \mapsto {X_i}']$ \;
        
        $\Gamma_i \gets (b_i \implies F')$ \;

      }
    }
  }

  \Return $\Gamma_n$ \;

\end{algorithm}

\begin{lemma}
  \label{pe:freevars}
  For all $1 \le j \le n$, the only free variables in $\Gamma_j$
  are of the form $b_k$, $X_k$, or ${X_k}'$ for some $1 \le k \le
  j$ such that $P_k \in LF(P_j)$.
\end{lemma}

\begin{proof}
  We proceed by induction on the iteration $i$ of the for loop
  and assume the claim holds for each $j < i$. Certainly, this is
  the case before the first iteration. And, for any $i$ and any
  $j < i$, the claim still holds after the $i$th iteration since
  $\Gamma_j$ is not modified. For $j = i$, if $P_i = 1$,
  $P_i = e \in E$, or $P_i = {P_k}^*$ for some $k$, then the only
  free variables in $\Gamma_i$ are $b_i$, $X_i$, and ${X_i}'$, so
  we are done. Otherwise, $P_i = P_j + P_k$ or
  $P_i = P_j \cdot P_k$ for some $j$ and $k$, and the free
  variables in $\Gamma_i$ are exactly those in $\Gamma_j$ or
  $\Gamma_k$ or $b_i, b_j, b_k, X_i, {X_i}', X_j, {X_j}', X_k$ or
  ${X_k}'$. By the induction hypothesis and the well-formedness
  of $P$, the claim holds for $i$.
\end{proof}

Let $V_j = \bigcup_{P_k \in LF(P_j)} \{b_k, X_k, {X_k}'\}$. Note
that by the above Lemma, if $P_j \in LF(P_i)$, then
$V_j \subseteq V_i$. As an abuse of notation, we write
$\exists V_j$ to denote
$\exists b_{k_1}, \dots, b_{k_l}, X_{k_1}, \dots, X_{k_l},
{X_{k_1}}', \dots, {X_{k_l}}'$, where
$LF(P_j) = \{P_{k_1}, \dots, P_{k_l}\}$.

\begin{lemma}
  \label{pe:ext}
  For all $1 \le j \le i \le n$, if $P_j \in LF(P_i)$, then for
  any formula $\varphi$ which does not contain any free variables
  in $V_i \setminus V_j$,
  $(\exists V_j. \Gamma_j \wedge \varphi) \iff (\exists
  V_i. \Gamma_i \wedge \varphi)$.
\end{lemma}

\begin{proof}
  Since $\Gamma_i \models \Gamma_j$ and $V_j \subseteq V_i$, the
  backwards direction is trivial. For the forwards direction,
  suppose there are some bindings to the variables in $V_j$ such
  that $\Gamma_j \wedge \varphi$. Then, extend those bindings to
  all variables in $V_i$ such that for all $b_k \not\in V_j$,
  $b_k \equiv false$. Clearly, we still have that
  $\Gamma_j \wedge \varphi$. Furthermore, consider the case that
  $\Gamma_i \models \psi$ but $\Gamma_j \not\models \psi$ for
  some $\psi$. Then, $\psi \equiv b_k \implies \psi'$ for some
  $\psi'$. Since $b_k \equiv false$, $\psi$ holds
  vacuously. Therefore, since $\Gamma_i \models \Gamma_j$, we
  must have that $\Gamma_i \wedge \varphi$ holds. 
\end{proof}

\begin{theorem}
  $Sat(\tf{P}) \iff Sat(\Gamma \wedge b_n)$
\end{theorem}

\begin{proof}
  We proceed by induction on the iteration $i$ of the for loop,
  with the hypothesis that before the $i$th iteration, for all
  $j < i$,
  $\tf{P_j} \equiv \exists V_j. \Gamma_j \wedge b_j \wedge X_j =
  X \wedge {X_j}' = X'$.

  Certainly this is true before the first iteration. Suppose this
  is true before the $i$th iteration of the loop. Clearly, the
  claim still holds for $j < i$ after this iteration. For
  $j = i$, let us consider each possible form of $P_i$
  separately:

  \begin{itemize}
  \item $P_i = 1$: We have that $\tf{1} \equiv X = X'$ and
    $\Gamma_i \equiv (b_i \implies X_i = {X_i}')$.  So,
    $\exists b_i, X_i, {X_i}'. \Gamma_i \wedge b_i \wedge X_i = X
    \wedge {X_i}' = X' \equiv X = X' \equiv \tf{P_i}$.

  \item $P_i = \tuple{u, v}$: We have that
    $\tf{\tuple{u, v}} = F_{\tuple{u, v}}(X, X')$ and
    $\Gamma_i \equiv (b_i \implies F_{\tuple{u, v}}(X_i,
    {X_i}'))$.  So,
    $\exists b_i, X_i, {X_i}'. \Gamma_i \wedge b_i \wedge X_i = X
    \wedge {X_i}' = X' \equiv F_{\tuple{u, v}}(X, X') \equiv
    \tf{P_i}$.
    
  \item $P_i = P_j + P_k$:
    $\tf{P_j+P_k} \equiv \tf{P_j} \vee \tf{P_k}$. By the
    induction hypothesis,
    $\tf{P_i} \equiv (\exists V_j. \Gamma_j \wedge b_j \wedge X_j
    = X \wedge {X_j}' = X') \vee (\exists V_k. \Gamma_k \wedge
    b_k\wedge X_k = X \wedge {X_k}' = X')$.

    By Lemma \ref{pe:ext}, this means
    $\tf{P_i} \equiv \exists V_i. \Gamma_i \wedge ((b_j \wedge
    X_j = X_i \wedge {X_j}' = {X_i}') \vee (b_k \wedge X_k = X_i
    \wedge {X_k}' = {X_i}')) \wedge X_i = X \wedge {X_i}' = X'
    \equiv \exists V_i. \Gamma_i \wedge b_i \wedge X_i = X \wedge
    {X_i}' = X'$ as desired.
    
  \item $P_i = P_j \cdot P_k$:
    $\tf{P_j \cdot P_k} \equiv \exists X''_1, X''_2. \tf{P_j}(X,
    X''_1) \wedge \tf{P_k}(X''_2, X') \wedge X''_1 = X''_2$. So,
    by the induction hypothesis,
    $\tf{P_i} \equiv \exists X''_1, X''_2. (\exists V_j. \Gamma_j
    \wedge b_j \wedge X_j = X \wedge {X_j}' = X''_1) \wedge
    (\exists V_k. \Gamma_k \wedge b_k \wedge X_k = X''_2 \wedge
    {X_k}' = X') \wedge X''_1 = X''_2$.

    Suppose $V_j \cap V_k \ne \emptyset$. Then, there must be
    some $P_l \in LF(P_j) \cap LF(P_k)$. But, this contradicts
    the well-formedness of $P$, from which we know that
    $LF(P_j) \cap LF(P_k) = \emptyset$. So,
    $V_j \cap V_k = \emptyset$. By Lemma \ref{pe:freevars}, this
    means there are no free variables present in both $\Gamma_j$
    and $\Gamma_k$. Therefore,
    \begin{align*}
      \tf{P_i}
      &\equiv \exists X''_1, X''_2. \exists V_i. \Gamma_j
        \wedge b_j \wedge X_j = X \wedge {X_j}' = X''_1 \wedge
        \Gamma_k \wedge b_k \wedge X_k = X''_2 \wedge {X_k}' = X'
        \wedge X''_1 = X''_2 \\
      &\equiv \exists V_i. \Gamma_j \wedge b_j
        \wedge X_j = X \wedge {X_j}' = X_k \wedge \Gamma_k \wedge
        b_k \wedge {X_k}' = X' \\
      & \equiv \exists V_i. \Gamma_i \wedge
        b_i \wedge X_i = X \wedge {X_i}' = X'
    \end{align*}

  \item $P_i = {P_j}^*$: We know by the induction hypothesis that
    $\tf{P_j} \equiv \exists V_j. \Gamma_j \wedge b_j \wedge X_j
    = X \wedge {X_j}' = X'$.  So,
    $F = star(\Gamma_j, b_j, \tuple{X_j, {X_j}'}) \equiv
    \tf{P_i}(X_j, {X_j}')$. Therefore,
    $F' = F[X_j \mapsto X_i, {X_j}' \mapsto {X_i}'] \equiv
    \tf{P_i}(X_i, {X_i}')$. So, since
    $\Gamma_i \equiv (b_i \implies F)$,
    $\exists b_i, X_i, {X_i}'. \Gamma_i \wedge b_i \wedge X_i = X
    \wedge {X_i}' = X' \equiv \tf{P_i}$.
  \end{itemize}
\end{proof}
